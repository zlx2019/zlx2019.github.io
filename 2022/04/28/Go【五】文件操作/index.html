Go【五】文件操作

发表于2022-04-28|更新于2023-03-15|Go基础
|字数总计:1.7k|阅读时长:6分钟|阅读量:

前言

一直以来都角色Go的文件IO操作非常简洁、好用,Java的文件操作太过于冗余和杂乱,一直没有深入研究使用,今天终于有时间好好把Go的相关文件IO操作使用并且记录一下~

1. 基础使用

1. 文件模式

在Golang中 通过io打开文件需要指定权限,权限有以下列举这些。

// Flags to OpenFile wrapping those of the underlying system. Not all
// flags may be implemented on a given system.
const (
  //  O_RDONLY, O_WRONLY, or O_RDWR 必选一项
  O_RDONLY int = syscall.O_RDONLY // 以只读方式打开文件
  O_WRONLY int = syscall.O_WRONLY // 以只写方式打开文件
  O_RDWR   int = syscall.O_RDWR   // 以可读可写方式打开文件
  O_APPEND int = syscall.O_APPEND // 写入数据时以追加形式,否则会覆盖文件原有数据
  O_CREATE int = syscall.O_CREAT  // 如果打开的文件不存在则新建文件
  O_EXCL   int = syscall.O_EXCL   // 和O_CREATE一起使用，表示创建文件时,该文件不能存在
  O_SYNC   int = syscall.O_SYNC   // 打开文件的同步io
  O_TRUNC  int = syscall.O_TRUNC  // 文件打开时裁剪文件
)

2. 创建文件

file, err := os.Create("zero日记.txt")
if err != nil {
  log.Fatalf("文件创建失败: %v", err.Error())
}
log.Println("文件创建成功,fileName:", file.Name())

文件如果存在,会直接覆盖原有文件,并且默认创建的文件会以O_RDWR|O_CREATE|O_TRUNC权限和0666权限。

func Create(name string) (*File, error) {
	return OpenFile(name, O_RDWR|O_CREATE|O_TRUNC, 0666)
}

如果想更友好的创建文件,我们则需要通过os.openFile()来指定模式。

//如果不存在则创建文件。O_CREATE
file, err := os.OpenFile("zero日记.txt",os.O_CREATE, 0600)

//创建文件,文件如果已存在则失败
file, err := os.OpenFile("zero日记.txt", os.O_CREATE|os.O_EXCL, 0600)

3. 裁剪文件

将一个文件内容裁剪到剩余指定的字节。

如裁剪一个文件到100个字节,如果文件本来就少于100个字节，则文件中原始内容得以保留，剩余的字节以null字节填充。如果文件本来超过100个字节，则超过的字节会被抛弃。 传入0则会清空文件。

将文件内容按字节裁剪,如文件内容如你好,裁剪文件内容剩余3字节,则剩你。

//txtVal:你好
err := os.Truncate("t1.txt", 3)
if err != nil {
  log.Fatalln(err.Error())
}
//你

4. 获取文件信息

fileInfo, err := os.Stat("t1.txt")
if err != nil {
  log.Fatalln("获取文件信息失败:", err.Error())
}
fmt.Println("文件名:", fileInfo.Name())
fmt.Println("文件字节大小:", fileInfo.Size())
fmt.Println("是否是文件夹:", fileInfo.IsDir())

fmt.Println("文件权限:", fileInfo.Mode())

5. 文件重命名和移动

//文件重命名,如果新的文件名称已存在,则会覆盖掉该已存在的文件
func fileRename() {
	err := os.Rename("new.txt", "d.txt")
	if err != nil {
		log.Fatalln("文件重命名失败: ", err.Error())
	}
}
//移动的文件夹必须存在
func fileRename() {
	err := os.Rename("d.txt", "temp/d.txt")
	if err != nil {
		log.Fatalln("文件移动失败: ", err.Error())
	}
}

6. 删除文件

//删除文件或空文件夹
//如果删除文件夹,文件夹内不能有任何文件
func removeFile() {
	err := os.Remove("temp/")
	if err != nil {
		log.Fatalln("删除文件失败:", err.Error())
	}
}

强制删除文件夹包括里边所有内容。

func removeFile() {
	err := os.RemoveAll("temp/")
	if err != nil {
		log.Fatalln("删除文件失败:", err.Error())
	}
}

7. 打开文件

  • 简单已只读方式打开

    func openFile() {
	file, err := os.Open("go.mod")
	if err != nil {
		log.Fatalln("打开文件失败:", err.Error())
	}
	//关闭文件
	file.Close()
}

  • 自定义打开权限

    func openFile() {
	//OpenFile 提供更多打开选项
	//最后一个参数是权限模式permission mode
	file, err := os.OpenFile("go.mod", os.O_RDWR, 0666)
	if err != nil {
		log.Fatalln("打开文件失败:", err.Error())
	}
	//关闭文件
	file.Close()
}

8. 文件是否存在

func fileExist() {
	_, err := os.Stat("go.mod")
	if err != nil {
		if os.IsNotExist(err) {
			log.Fatalln("该文件不存在!")
		}
	}
}

9. 检查文件读写权限

通过其他角色创建了一个文件/usr/local/t.txt

func checkFilePermission() {
	_, err := os.OpenFile("/usr/local/t.txt", os.O_RDWR, 0666)
	if err != nil {
		//判断是否是 权限不匹配error,也可能是文件不存在error
		if os.IsPermission(err) {
			log.Fatalln("文件权限不匹配,无法操作该文件:" + err.Error())
		}
	}
}

10. 修改文件权限、拥有者、时间戳

修改者必须拥有权限才能执行。

func modifyFilePermission() {
	//修改文件权限,以linux风格
	err := os.Chmod("/usr/local/t.txt", 0777)

	if err != nil {
		log.Fatalln(err.Error())
	}

	//修改文件所有者为当前登陆者
	err = os.Chown("/usr/local/t.txt", os.Getuid(), os.Getuid())

	if err != nil {
		log.Fatalln(err.Error())
	}

	//修改文件的时间戳
	now := time.Now().Add(48 * time.Hour)
	lastAccessTime := now
	lastModifyTime := now
	err = os.Chtimes("/usr/local/t.txt", lastAccessTime, lastModifyTime)

	if err != nil {
		log.Fatalln(err.Error())
	}
}

2. 文件写入

1. 原生写入

func writeFile() {
	//打开文件,如果不存在则创建,追加写入数据
	file, err := os.OpenFile("book.md", os.O_CREATE|os.O_RDWR|os.O_APPEND, 0666)
	if err != nil {
		log.Fatalln("打开文件失败:", err.Error())
	}
	defer file.Close() //关闭文件

	//写入内容
	body := []byte("Hello,World")

	//写入到文件中
	//count: 写入的字节数
	//err:  错误
	count, err := file.Write(body)
	if err != nil {
		log.Fatalln("写入数据失败:", err.Error())
	}
	log.Println("写入数据成功:", count)
}

2. ioutil写入

Go提供了许多工具包,可以让我们更方便的进行IO操作。

func writeFileByUtil() {
	err := ioutil.WriteFile("book.txt", []byte("ioUtil快速写入~"), 0666)
	if err != nil {
		log.Fatalln("写入数据失败!")
	}
}

此方式会直接覆盖原有数据。

3. bufio写入

bufio包提供了带缓存功能的writer，所以你可以在写字节到硬盘前使用内存缓存。当你处理很多的数据很有用，因为它可以节省操作硬盘I/O的时间。在其它一些情况下它也很有用，比如你每次写一个字节，把它们攒在内存缓存中，然后一次写入到硬盘中，减少硬盘的磨损以及提升性能。

func writeFileByBuf() {
	//打开文件:只写、追加
	file, _ := os.OpenFile("book.txt", os.O_WRONLY|os.O_APPEND, 0666)

	defer file.Close()

	//创建写入缓存块
	bufWriter := bufio.NewWriter(file)

	//向缓存写入数据
	writerLen, _ := bufWriter.WriteString("今天天气真不错哟!")
	log.Println("写入数据:", writerLen)

	//获取缓存中的字节数
	bytes := bufWriter.Buffered()
	log.Println("缓存中的字节数:", bytes)

	//还有多少字节可用
	activeBytes := bufWriter.Available()
	log.Println("可用字节:", activeBytes)

	//将缓存的数据 写入到文件
	bufWriter.Flush()
}

3. 文件读取

1. 快读文件

读取文件内容到slice中。

func readFileBy() {
	body, _ := ioutil.ReadFile("fle/base/base.go")
	log.Println(string(body))
}

2. ioutil读取

读取文件全部数据。

func readFile() {
	file, _ := os.OpenFile("file/base/base.go", os.O_RDONLY, 0666)
	defer file.Close()

	body, err := ioutil.ReadAll(file)
	if err != nil {
		log.Fatalln("读取错误:", err.Error())
	}
	log.Println(string(body))
}

4. 文件copy

//文件复制
func fileCopy() {
	//打开要复制的文件
	file, _ := os.Open("book.txt")
	defer file.Close()

	//创建新文件
	copyFile, _ := os.OpenFile("book_copy.txt", os.O_CREATE|os.O_EXCL|os.O_RDWR, 0666)
	defer copyFile.Close()

	//copy文件内容到新文件
	bytes, _ := io.Copy(copyFile, file)
	log.Println("copy成功:", bytes)

	//将文件内容刷新到硬盘
	copyFile.Sync()
}
