Java【高并发编程-CompletableFuture】

发表于2022-04-10|更新于2022-04-12
|字数总计:2.7k|阅读时长:11分钟|阅读量:

前言

使用Future获得异步执行结果时，要么调用阻塞方法get()，要么轮询看isDone()是否为true，这两种方法都不是很好，因为主线程也会被迫等待。

从Java 8开始引入了CompletableFuture，它针对Future做了改进，可以传入回调对象，当异步任务完成或者发生异常时，自动调用回调对象的回调方法。

Demo

工具类

public class SmallTool {
    public static void sleepMillis(long millis){
        try {
            Thread.sleep(millis);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    public static void printTimeAndThread(String tag){
        String result = new StringJoiner("\t|\t")
                .add(String.valueOf(System.currentTimeMillis()))
                .add(String.valueOf(Thread.currentThread().getId()))
                .add(String.valueOf(Thread.currentThread().getName()))
                .add(tag).toString();
        System.out.println(result);
    }
}

1.supplyAsync

开启一个异步线程执行任务

public class CompletableFuture01 {
    public static void main(String[] args) {
        SmallTool.printTimeAndThread("小白进入餐厅");
        SmallTool.printTimeAndThread("小白点了一碗米饭、一个番茄炒蛋");

        //开启异步任务执行
        CompletableFuture<String> task  = CompletableFuture.supplyAsync(()->{
           SmallTool.printTimeAndThread("厨师炒菜");
           SmallTool.sleepMillis(200);
           SmallTool.printTimeAndThread("厨师做饭");
           SmallTool.sleepMillis(100);
           return "番茄炒蛋+ 米饭";
        });
        SmallTool.printTimeAndThread("小白点完餐后开始打王者...");
        SmallTool.printTimeAndThread(String.format("饭好了,小白开始吃饭:%s",task.join()));

    }
}

2.thenCompose(串行)

连接多个异步任务,并且顺序执行

/**
 * CompletableFuture 入门Demo2
 *  顺序执行多个异步任务
 * @author lx Zhang.
 * @date 2021/3/26 10:28 下午
 */
public class CompletableFuture02 {
    public static void main(String[] args) {
        SmallTool.printTimeAndThread("小白进入餐厅");
        SmallTool.printTimeAndThread("小白点了一碗米饭、一个番茄炒蛋");

        //开启异步任务执行 CompletableFuture.supplyAsync
        CompletableFuture<String> task  = CompletableFuture.supplyAsync(()->{
           SmallTool.printTimeAndThread("厨师炒菜");
           SmallTool.sleepMillis(200);
           return "番茄炒蛋";
           //上一个任务结束后将结果传递给下一个异步任务
        }).thenCompose(dish-> CompletableFuture.supplyAsync(()->{
            SmallTool.printTimeAndThread("服务员接过菜后去打饭");
            SmallTool.sleepMillis(200);
            return dish + "米饭";
        }));
        SmallTool.printTimeAndThread("小白点完餐后开始打王者...");
        SmallTool.printTimeAndThread(String.format("饭好了,小白开始吃饭:%s",task.join()));

    }
}

3.thenCombine(并行)

将两个异步任务并行,都执行完之后,合并结果

/**
 * CompletableFuture 入门Demo3
 *  将两个异步任务并行,都执行完之后,合并结果
 * @author lx Zhang.
 * @date 2021/3/26 10:28 下午
 */
public class CompletableFuture03 {
    public static void main(String[] args) {
        SmallTool.printTimeAndThread("小白进入餐厅");
        SmallTool.printTimeAndThread("小白点了一碗米饭、一个番茄炒蛋");

        //创建任务1: 厨师炒菜
        CompletableFuture<String> job = CompletableFuture.supplyAsync(()->{
           SmallTool.printTimeAndThread("厨师炒菜");
           SmallTool.sleepMillis(200);
           return "番茄炒蛋";
        }).thenCombine(CompletableFuture.supplyAsync(()->{
            //通过thenCombine()函数 再开启一个异步任务,两个任务是并行的
            //需要传入两个参数,p1:要并行的任务,p2:任务都执行完后的回调函数
            SmallTool.printTimeAndThread("服务员蒸大米饭");
            SmallTool.sleepMillis(200);
            return "大米饭";
        }),(dish,rice)->{
            //合并任务结果
            SmallTool.printTimeAndThread("服务员将菜盛到米饭上");
            return dish + rice;
        });
        SmallTool.printTimeAndThread("小白点完餐后开始打王者...");
        SmallTool.printTimeAndThread(String.format("饭好了,小白开始吃饭:%s",job.join()));
    }
}

4.thenApplyAsync

将两个异步任务串行化执行 最后完成的任务结果作为返回值

/**
 * CompletableFuture 入门Demo3
 *  将两个异步任务串行化执行 最后完成的任务结果作为返回值
 * @author lx Zhang.
 * @date 2021/3/26 10:28 下午
 */
public class CompletableFuture04 {
    public static void main(String[] args) {
        SmallTool.printTimeAndThread("小白吃完了饭");
        SmallTool.printTimeAndThread("小白要求开发票!");

        //创建任务1: 服务员收款
        CompletableFuture<String> job = CompletableFuture.supplyAsync(()->{
            SmallTool.printTimeAndThread("服务员1号收款500元");
            SmallTool.sleepMillis(200);
            return "500";
        }).thenApplyAsync(money->{
            //执行完上一个任务后,再执行该任务
            SmallTool.printTimeAndThread("服务员2号开发票 面额:" + money);
            SmallTool.sleepMillis(200);
            return money + "发票";
        });
        SmallTool.printTimeAndThread("小白接到朋友电话...");
        SmallTool.printTimeAndThread(String.format("小白拿到发票,回家",job.join()));
    }
}

5.applyToEither

同时执行两个异步任务,哪个任务先执行完成,就把哪个任务的结果作为返回值,并且做异常处理

/**
 * CompletableFuture 入门Demo3
 *  场景:
 *      同时执行两个异步任务,哪个任务先执行完成,就把哪个任务的结果作为返回值
 *      并且做异常处理
 * @author lx Zhang.
 * @date 2021/3/26 10:28 下午
 */
public class CompletableFuture05 {
    public static void main(String[] args) {
        SmallTool.printTimeAndThread("小白来到公交车站");
        SmallTool.printTimeAndThread("等775 or 889路公交车");
        SmallTool.printTimeAndThread("等待中....");

        //创建异步任务1 775路车
        CompletableFuture<String> job = CompletableFuture.supplyAsync(()->{
            SmallTool.printTimeAndThread("775路车正在赶来的路上...");
            SmallTool.sleepMillis(500);
            return "775";
            //创建异步任务2 889路车
        }).applyToEither(CompletableFuture.supplyAsync(()->{
            SmallTool.printTimeAndThread("889路车正在赶来的路上...");
            SmallTool.sleepMillis(400);
            return "889";
        }), firstResult->{
            return firstResult;
        });
        SmallTool.printTimeAndThread("小白坐车回家:"+ job.join());
    }
}

public class CompletableFuture05 {
    public static void main(String[] args) {
        SmallTool.printTimeAndThread("小白来到公交车站");
        SmallTool.printTimeAndThread("等775 or 889路公交车");

            SmallTool.sleepMillis(500);
            return "775";
            //创建异步任务2 889路车
        }).applyToEither(CompletableFuture.supplyAsync(()->{
            SmallTool.printTimeAndThread("889路车正在赶来的路上...");
            SmallTool.sleepMillis(400);
            return "889";
        }), firstResult->{
            //模拟异常出现
            if (firstResult.equals("889")){
                throw new RuntimeException("出车祸了...");
            }
            return firstResult;
        }).exceptionally(throwable -> {
            SmallTool.printTimeAndThread(throwable.getMessage());
            SmallTool.printTimeAndThread("小白呼叫出租车");
            return "出租车";
        });
        SmallTool.printTimeAndThread("小白坐车回家:"+ job.join());
    }
}

API

package com.zero.java8.future;

import java.util.Arrays;
import java.util.List;
import java.util.Optional;
import java.util.concurrent.*;
import java.util.stream.Collectors;

/**
 * CompletableFuture
 *
 * @author Zero.
 * @date 2022/3/23 3:08 下午
 */
public class Main {
    //自定义线程池
    private static ExecutorService executor = new ThreadPoolExecutor(4, 8, 60, TimeUnit.SECONDS,
            new LinkedBlockingDeque<>(), r -> {
                Thread thread = new Thread(r);
                thread.setDaemon(false);
                return thread;
            });

    public static void main(String[] args) {
        testSupplyAsync();
        testThenApplyAsync();
        testHandle();
        testThenRun();
        testApplyToEither();

        //stop
        executor.shutdown();
    }

    /**
     * 1.supplyAsync()
     * CompletableFuture创建异步任务
     * 通过CompletableFuture.supplyAsync() 创建一个异步任务,参入一个Supplier接口-->实现异步任务业务
     * 通过通过CompletableFuture.whenComplete() 获取该任务结果,参数为(任务结果、异常信息)
     */
    private static void testSupplyAsync(){
        CompletableFuture.supplyAsync(()->{
            return "get string";
        },executor).whenComplete((result,exception)->{
            Optional.ofNullable(result).ifPresent(System.out::println);
        });
    }

    /**
     * 2.thenApply | thenApplyAsync
     * 通过CompletableFuture.thenApply() 获取上一步任务结果,最终要处理后再次返回抛出
     */
    private static void testThenApplyAsync(){
        CompletableFuture<String> apply = CompletableFuture.supplyAsync(() -> "result")
                .thenApplyAsync(result -> {
                    return result + "Hello";
                });
    }

    /**
     * 3. handle
     * 与thenApply类似, 获取上一步任务的结果,并且能捕获到异常
     */
    private static void testHandle(){
        CompletableFuture.supplyAsync(()-> "result")
                .handle((result,exception)-> {
                    if (exception != null){
                        return "发生了异常!";
                    }
                    result = result + "Hello";
                    return result;
                });
    }

    /**
     * 4. thenAccept
     * 获取到任务结果,没有出参，只能对任务结果进行消费,通常用于任务结束处理函数
     */
    private static void testThenAccept(){
        CompletableFuture.supplyAsync(()-> "result")
                .thenAccept(result->{
                    System.out.println("任务已完成,结果:" + result);
                });
    }

    /**
     * 5.whenCompleteAsync | whenComplete
     * 通常用于所有任务结束后执行的方法,能够获取到结果数据与异常信息
     */
    private static void testWhenCompleteAsync(){
        CompletableFuture.supplyAsync(()-> "result")
                .thenApply(s-> s + "Hello")
                .whenCompleteAsync((result,e)->{
                    System.out.println("任务完成 结果:" + result);
                });
    }

    /**
     * 6.thenRun
     * 整个CompletableFuture流程执行完成后回调的方法,没有入参,只能作为回调通知方法
     */
    private static void testThenRun(){
        CompletableFuture.supplyAsync(()-> "result")
                .thenApply(result-> result + "hello")
                .whenCompleteAsync((result,e)->{
                    System.out.println("任务完成 结果:" + result);
                })
                .thenRun(()-> System.out.println("任务执行完成"));
    }

    /**
     * 7. thenCompose
     * 组合式CompletableFuture,可以将上一个CompletableFuture执行的结果 交给下一个CompletableFuture进行使用
     * thenCompose的入参1 为上个CompletableFuture的返回结果
     *
     */
    private static void testThenCompose(){
        CompletableFuture.supplyAsync(()-> "result")
                .thenCompose(result-> CompletableFuture.supplyAsync(()-> result + "append Value"))
                .thenAccept(endResult-> System.out.println("任务执行完成 最终结果:" + endResult));
    }

    /**
     * 8.thenCombine
     * 将两个CompletableFuture结果获取到 进行处理
     */
    private static void testThenCombine(){
        CompletableFuture.supplyAsync(()-> 100)
                .thenCombine(CompletableFuture.supplyAsync(()-> 200),(result1,result2)-> {
                   return result1 + result2;
                }).thenAccept(System.out::println);
    }

    /**
     * 9. thenAcceptBoth | thenAcceptBothAsync
     * 开启一个新的CompletableFuture 并且将 两个任务结果获取到进行处理,无返回值,只能用来消费结果
     */
    private static void testThenAcceptBoth(){
        CompletableFuture.supplyAsync(()-> 100)
                .thenAcceptBothAsync(CompletableFuture.supplyAsync(()-> 200),(v1,v2)->{
                    System.out.println("任务一结果:" + v1);
                    System.out.println("任务二结果:" + v2);
                });
    }

    /**
     * 10. runAfterBothAsync
     * 该方法入参一个新的任务、一个Runnable接口
     * 只有本身任务与新任务都结束后 才会执行Runnable,并且是没有返回值的
     */
    private static void testRunAfterBoth(){
        CompletableFuture.supplyAsync(()-> 1001)
                .runAfterBothAsync(CompletableFuture.supplyAsync(()-> 2001),()->{
                    System.out.println("两个任务执行完成了,但是没有返回值哦~");
                });
    }

    /**
     * 11. applyToEither
     * 入参为一个新的任务 一个Function方法,最先完成的任务结果将会传入到Function
     */
    private static void testApplyToEither(){
        CompletableFuture<Integer> task1 = CompletableFuture.supplyAsync(() ->{
            try {
                TimeUnit.SECONDS.sleep(1);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            return 1;
        });
        CompletableFuture<Integer> task2 = CompletableFuture.supplyAsync(() -> 2);
        task1.applyToEither(task2,(result)->{
            System.out.println("已经有一个任务完成 结果:" + result);
            return result;
        });
    }

    /**
     * 12. allOf
     * 入参为一个CompletableFuture 序列 只有所有的CompletableFuture都结束后才会执行后续操作
     *
     * 13.anyOf
     * 入参同样 只要有一个任务完成 就执行后续操作
     */
    private static void testAllOf(){
        List<Integer> list = Arrays.asList(1, 2, 3, 4, 5);
        CompletableFuture[] taskArr = list.stream().map(i -> CompletableFuture.supplyAsync(() -> i))
                .collect(Collectors.toList())
                .toArray(new CompletableFuture[list.size()]);
        CompletableFuture.allOf(taskArr).thenRun(()-> System.out.println("所有任务已经执行完成!"));
    }

    /**
     * 需求 从db中获取一组Double序列,将每个元素翻十倍，然后汇总
     */
    private static void testThenCompletable() {
        List<Integer> dbValue = Arrays.asList(12, 23, 56, 213, 64,12,53,21,56,21);
        Integer sum = dbValue.stream().map(val -> CompletableFuture.supplyAsync(() -> val))
                .map(future -> future.thenApply(val -> val * 10))
                .map(CompletableFuture::join)
                .reduce(0, Integer::sum);
        System.out.println(sum);
    }

}
