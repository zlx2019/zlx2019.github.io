Unity物体移动

发表于2022-05-30|更新于2022-06-08|Unity
|字数总计:1.1k|阅读时长:4分钟|阅读量:

前言

关于Unity中的物体移动方式有很多,我自己所知的大概有七八种,分为非物理移动和物理移动。

非物理移动

// 移动速度
private float speed = 5;

1. 直接换算Position 方式一

直接根据gameObject的position计算变换。

void Update()
{
  MovementByTransform();
}

/// <summary>
/// 方式一: 直接变换 gameObject的transform的position属性: X、Y、Z轴
/// </summary>
void MovementByTransform()
{
  // 按W键--前进 增加Z轴,其他轴不变
  if (Input.GetKey(KeyCode.W))
  {
    // 移动速度需要 * Time.deltaTime,这样才能保证不同及机器上移动速度相同
    transform.localPosition = new Vector3(transform.localPosition.x, transform.localPosition.y, transform.localPosition.z + speed * Time.deltaTime);
  }
  // 按S键--后退 递减Z轴,其他轴不变
  if (Input.GetKey(KeyCode.S))
  {
    // 移动速度需要 * Time.deltaTime,这样才能保证不同及机器上移动速度相同
    transform.localPosition = new Vector3(transform.localPosition.x, transform.localPosition.y, transform.localPosition.z - speed * Time.deltaTime);
  }
}

左右移动就不多写了,道理一样的~

此方式缺点: 整体移动观感较为僵硬,输入受帧率影响,代码过于冗余。

2. 方式二

通过transform的Translate()方法移动。

/// <summary>
/// 方式二 通过transform的Translate()方法移动
/// </summary> 
void MovementByTranslate()
{
  float x = Input.GetAxis("Horizontal"); // 获取水平方向的输入 a、d、←、→
  float y = Input.GetAxis("Vertical"); //获取垂直方向的输入 w、s、↑、↓
  //将移动上下左右输入 构建成一个新的坐标值
  Vector3 move = new Vector3(x * speed * Time.deltaTime, 0, y * speed * Time.deltaTime);
  //使当前物体 移动到 move坐标
  transform.Translate(move);
}

效果如下:

此方式优点: 有一定的惯性,不受帧率影响

3. 方式三

这种方式是结合第一种和第二种改编,基本上类似.

/// <summary>
/// 方式三: 通过transform的Translate方法 和 Vector3内置的坐标方法移动
/// </summary>
void MovementByTheThird()
{
  if (Input.GetKey(KeyCode.W))
  {
    transform.Translate(Vector3.forward * speed * Time.deltaTime);
  }
  if (Input.GetKey(KeyCode.S))
  {
    transform.Translate(Vector3.back * speed * Time.deltaTime);
  }
  if (Input.GetKey(KeyCode.A))
  {
    transform.Translate(Vector3.left * speed * Time.deltaTime);
  }
  if (Input.GetKey(KeyCode.D))
  {
    transform.Translate(Vector3.right * speed * Time.deltaTime);
  }
}

这里就不再演示,效果基本上一致。

4. 基于坐标点移动

基于指定的position坐标进行移动。

private float speed = 2f;//移动速度
public Transform start;// 起始坐标 = 当前物体的Transform
public Transform end;//目的地坐标
private float startTime;//开始移动时间
private float btDistance; //起始位置到目的位置的间距
void Start()
{
  // 记录开始移动时间
  startTime = Time.time;
  // 获取两个坐标之间的距离间距
  btDistance = Vector3.Distance(start.position, end.position);
}
// Update is called once per frame
void Update()
{
  MoveByLerp();
}

void MoveByLerp()
{
  //已经移动的时间 * 移动速度 = 已经移动的路程
  float moved = (Time.time - startTime) * speed * Time.deltaTime; 
  //已经移动的路程 / 总路程 = 已经移动完成的百分比
  float fractionOfBtDistance = moved / btDistance; 
  //让当前物体的位置 = 已经移动到的百分比位置
  transform.position = Vector3.Lerp(start.position, end.position, fractionOfBtDistance);
}

5. 基于坐标平滑阻尼

基于指定的时间,让物体平滑移动到指定的位置。

public Transform target; //目标点
public Vector3 temp = Vector3.zero;
private float smoothTime = 5f; //移动到目标点所需要的时间
void Update()
{
  // MoveByLerp();
  MoveBySmoothDamp();
}
void MoveBySmoothDamp()
{
  //Vector3.SmoothDamp(起始坐标, 目标点坐标, ref temp, 到达目标点所需要的时间);
  transform.position = Vector3.SmoothDamp(transform.position, target.position, ref temp, smoothTime);
}

物理移动

1. 刚体移动-Velocity

直接增加刚体的速度。

private float speed = 1f;//移动速度
private Rigidbody rb;//物体刚体

void Start()
{
  rb = GetComponent<Rigidbody>();//获取当前物体的刚体
}

// Update is called once per frame
void Update()
{
  rbMove1();
}

/// <summary>
/// 刚体移动方式一:
/// 增加刚体的velocity速度属性 X、Y、Z
/// </summary>
void rbMove1()
{
  if (Input.GetKey(KeyCode.W))
  {
    //前进 增加Z轴速度
    rb.velocity += new Vector3(0,0,speed);
  }
  if (Input.GetKey(KeyCode.S))
  {
    //后退 减少Z轴速度
    rb.velocity -= new Vector3(0,0,speed);
  }
  if (Input.GetKey(KeyCode.D))
  {
    //向右移动 增加X轴速度
    rb.velocity += new Vector3(speed,0,0);
  }
  if (Input.GetKey(KeyCode.A))
  {
    //向左移动 减少X轴速度
    rb.velocity -= new Vector3(speed,0,0);
  }
}

2. 方式二

private void FixedUpdate()
{
  rbMove2();
}

/// <summary>
/// 刚体移动方式二:
/// 通过刚体的MovePosition()方法移动 注意该方式适合放在FixedUpdate中
/// </summary>
void rbMove2()
{
  float h = Input.GetAxis("Horizontal") * speed * Time.fixedTime;
  float v = Input.GetAxis("Vertical") * speed * Time.fixedTime;
  rb.MovePosition(transform.position + new Vector3(h,0,v));
}

3. 方式三

private float speed = 50;//移动速度 这种方式速度需要设置高一点
private Rigidbody rb;//物体刚体

private void FixedUpdate()
{
  rbMove3();
}
void rbMove3()
{
  if (Input.GetKey(KeyCode.W))
  {
    rb.AddForce(new Vector3(0,0,speed));
  }
  if (Input.GetKey(KeyCode.S))
  {
    rb.AddForce(new Vector3(0,0,-speed));
  }
  if (Input.GetKey(KeyCode.A))
  {
    rb.AddForce(new Vector3(-speed,0,speed));
  }
  if (Input.GetKey(KeyCode.D))
  {
    rb.AddForce(new Vector3(speed,0,0));
  }
}
