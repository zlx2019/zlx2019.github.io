前言

Kratos是哔哩哔哩开源的一套轻量级Go微服务框架,包含大量微服务相关框架及工具。

名字来源于:《战神》游戏以希腊神话为背景，讲述奎托斯（Kratos）由凡人成为战神并展开弑神屠杀的冒险经历。

1. 环境准备

首先,需要安装好对应的依赖环境及工具:

  • go
  • protoc
  • Protoc-gen-go

以上环境如何安装本章不再讲述,下面来安装一下Kratos的终端工具。三种方式选其一即可。

  1. go get安装

    go get -u github.com/go-kratos/kratos/cmd/kratos/v2@latest

  2. go install 安装

    go install github.com/go-kratos/kratos/cmd/kratos/v2
# go 1.16版本以上需要指定版本号或使用最新版
go install github.com/go-kratos/kratos/cmd/kratos/v2@latest

  3. 源码编译安装

    git clone https://github.com/go-kratos/kratos
cd kratos
make install

2. helloworld

下面创建一个简单的helloworld项目体验一下。通过kratos命令创建项目基础脚手架。

kratos new <项目名称> -r <指定脚手架仓库源>

默认会拉取官方的 kratos-layout,如果没有科学上网可以选用码云的仓库。

kratos new helloworld -r https://gitee.com/go-kratos/kratos-layout.git

也可以使用自定义的末班

kratos new helloworld -r xxx-layout.git
🍺 Project creation succeeded helloworld
💻 Use the following command to start the project 👇:

$ cd helloworld
$ go generate ./...
$ go build -o ./bin/ ./... 
$ ./bin/helloworld -conf ./configs

			🤝 Thanks for using Kratos
	📚 Tutorial: https://go-kratos.dev/docs/getting-started/start

如上显示则表示项目已经构建完成,接下来进入项目。

下载所需依赖。

go mod tidy

然后运行项目。

kratos run
INFO msg=config loaded: config.yaml format: yaml
INFO msg=[gRPC] server listening on: [::]:9000
INFO msg=[HTTP] server listening on: [::]:8000

可以看到,项目已经跑起来了,http与grpc服务分别占用9000与8000端口。

接下来通过http请求测试下接口。

curl http://localhost:8000/helloworld/kratos
{"message":"Hello kratos"}

至此项目构建完成~

3. 项目结构详解

  • api

    该包内主要是我们自定义的Proto文件以及Proto文件所生成的go文件。

  • cmd

    项目的启动入口

  • configs

    配置文件目录

  • internal

    业务目录,业务代码通常都定义在这里。

    • biz: 接口层
    • conf: 配置
    • data: 数据访问层
    • server: 注册服务
    • service: 服务层

  • third_party

    官方的proto文件

  • Makefile
    该文件中有些常用的生成命令,大多数情况下我们都需要依赖这些命令。

4. 自定义API

创建proto

在api目录的v1目录中创建一个新的user包,在包内创建user.proto文件。

//指定protobuf 的版本
syntax = "proto3";

//导入其他proto文件
import "google/api/annotations.proto";

//指定生成的go文件的package,不能乱改,脚手架会基于这个命名规则生成文件
option go_package = "helloworld/api/helloworld/v1/user;v1";

//Java 生成规则
option java_multiple_files = true;
option java_package = "dev.kratos.api.helloworld.v1";
option java_outer_classname = "HelloworldProtoV1";

///定义 API接口
service User{

  //定义RPC接口
  rpc UserList(UserRequest)returns (UserReply){
    //定义该RPC接口对应的Http接口 Get请求
    option(google.api.http) = {
      get: "/api/user/list",
    };
  }

  //定义根据ID查询用户接口
  rpc UserOne(UserRequest) returns (UserReply){
    option(google.api.http) = {
      get: "/api/user/{id}"
    };
  }

  //定义新增用户接口 Post请求
  rpc AddUser(UserRequest)returns(UserReply){
    option(google.api.http) = {
      post: "/api/user",
      body: "*" //表示将请求body映射到UserRequest中
    };
  }

  //定义修改用户接口 Put请求
  rpc UpdateUser(UserRequest)returns(UserReply){
    option(google.api.http) = {
      put: "/api/user",
      body: "*"
    };
  }

  //定义删除用户接口 Put请求
  rpc DeleteUser(UserRequest)returns(UserReply){
    option(google.api.http) = {
      delete: "/api/user",
    };
  }
}

//接口请求参数体
message UserRequest{
  string id = 1;
  string username = 2;
  string password = 3;
}

//接口返回响应体
message UserReply{
  int32 code = 1;
  string message = 2;
  repeated string list = 3;
}

make api

很明显,这写创建好的方法都是我们在user.proto中定义好的接口,接下来我们简单来实现一下。

package service

import (
	"context"
	"log"

	pb "helloworld/api/helloworld/v1/user"
)

type UserService struct {
	pb.UnimplementedUserServer
}

func NewUserService() *UserService {
	return &UserService{}
}

var reply *pb.UserReply = &pb.UserReply{Code: 0, Message: "执行成功", List: []string{"data1", "data2"}}

func (s *UserService) UserList(ctx context.Context, req *pb.UserRequest) (*pb.UserReply, error) {
	log.Println("执行 UserList接口")
	log.Println("username:" + req.GetUsername())
	log.Println("password:" + req.GetPassword())
	return reply, nil
}
func (s *UserService) UserOne(ctx context.Context, req *pb.UserRequest) (*pb.UserReply, error) {
	log.Println("执行 UserOne 接口")
	log.Println("id:" + req.GetId())
	return reply, nil
}
func (s *UserService) AddUser(ctx context.Context, req *pb.UserRequest) (*pb.UserReply, error) {
	log.Println("执行 AddUser 接口")
	log.Println("username:" + req.GetUsername())
	log.Println("password:" + req.GetPassword())
	return reply, nil
}
func (s *UserService) UpdateUser(ctx context.Context, req *pb.UserRequest) (*pb.UserReply, error) {
	log.Println("执行 UpdateUser 接口")
	log.Println("username:" + req.GetUsername())
	log.Println("password:" + req.GetPassword())
	return reply, nil
}
func (s *UserService) DeleteUser(ctx context.Context, req *pb.UserRequest) (*pb.UserReply, error) {
	log.Println("执行 DeleteUser 接口")
	log.Println("username:" + req.GetUsername())
	log.Println("password:" + req.GetPassword())
	return reply, nil
}

注册服务

修改internal/service/service.go文件

var ProviderSet = wire.NewSet(NewGreeterService, NewUserService)
//将UserService服务注册

修改internal/server/http.go文件

package server

import (
	"github.com/go-kratos/kratos/v2/log"
	"github.com/go-kratos/kratos/v2/middleware/recovery"
	"github.com/go-kratos/kratos/v2/transport/http"
	v1 "helloworld/api/helloworld/v1"
	user "helloworld/api/helloworld/v1/user"
	"helloworld/internal/conf"
	"helloworld/internal/service"
)

// NewHTTPServer new a HTTP server.
func NewHTTPServer(c *conf.Server, greeter *service.GreeterService, logger log.Logger) *http.Server {
	var opts = []http.ServerOption{
		http.Middleware(
			recovery.Recovery(),
		),
	}
	if c.Http.Network != "" {
		opts = append(opts, http.Network(c.Http.Network))
	}
	if c.Http.Addr != "" {
		opts = append(opts, http.Address(c.Http.Addr))
	}
	if c.Http.Timeout != nil {
		opts = append(opts, http.Timeout(c.Http.Timeout.AsDuration()))
	}
	srv := http.NewServer(opts...)
	v1.RegisterGreeterHTTPServer(srv, greeter)
	//注册userService 注册到HTTP服务中
	user.RegisterUserHTTPServer(srv, service.NewUserService())
	return srv
}

修改internal/server/grpc.go文件

package server

import (
	"github.com/go-kratos/kratos/v2/log"
	"github.com/go-kratos/kratos/v2/middleware/recovery"
	"github.com/go-kratos/kratos/v2/transport/grpc"
	v1 "helloworld/api/helloworld/v1"
	user "helloworld/api/helloworld/v1/user"
	"helloworld/internal/conf"
	"helloworld/internal/service"
)

// NewGRPCServer new a gRPC server.
func NewGRPCServer(c *conf.Server, greeter *service.GreeterService, logger log.Logger) *grpc.Server {
	var opts = []grpc.ServerOption{
		grpc.Middleware(
			recovery.Recovery(),
		),
	}
	if c.Grpc.Network != "" {
		opts = append(opts, grpc.Network(c.Grpc.Network))
	}
	if c.Grpc.Addr != "" {
		opts = append(opts, grpc.Address(c.Grpc.Addr))
	}
	if c.Grpc.Timeout != nil {
		opts = append(opts, grpc.Timeout(c.Grpc.Timeout.AsDuration()))
	}
	srv := grpc.NewServer(opts...)
	v1.RegisterGreeterServer(srv, greeter)
	//将userService注册到GRPC服务中
	user.RegisterUserServer(srv, service.NewUserService())
	return srv
}
