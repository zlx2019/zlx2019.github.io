Go【三】反射

发表于2021-08-21|更新于2023-03-01|Go
|字数总计:762|阅读时长:3分钟|阅读量:

前言

和Java语言一样，Go也实现运行时反射，这为我们提供一种可以在运行时操作任意类型对象的能力。比如我们可以查看一个接口变量的具体类型，看看一个结构体有多少字段，如何修改某个字段的值等等。

Go反射能力来源于reflect包

1. 断言

通过断言我们能够获取到 一个insterface具体的类型,如下

创建两个类似结构体

type Student struct {
	nikeName string
	age int8
}

type Teacher struct {
	nikeName string
	age int8
}

创建一个函数,入参为任意类型

func ShowName(any interface{}) {
	//无法操作....
}

假如现在我们使用Student作为参数入参

func main() {
	s := Student{nikeName: "小张同学"}
	ShowName(s)
}

对于ShowName函数执行过程中,它并不直到具体参数是什么,无法执行具体的操作,这个时候我们可以利用断言,然后类型转换,才能做具体的操作

func ShowName(any interface{}) {
	//通过 any.(type) 获取到 any的类型,通常和switch为固定写法
	//通过 any.(T) 将any强行断言为T类型
	switch any.(type){
	case Student:
		fmt.Println("any is Student",any.(Student).nikeName)
	case Teacher:
		fmt.Println("any is Teacher",any.(Teacher).nikeName)
	}
}
func main() {
	s := Student{nikeName: "小张同学"}
	ShowName(s)
	t := Teacher{nikeName: "刘老师"}
	ShowName(t)
}

查看执行结果

any is Student 小张同学
any is Teacher 刘老师

2. 反射

通过反射获取一个未知参数的类型与值

func CheckAny(any interface{}) {
	t := reflect.TypeOf(any) //获取any的类型
	v := reflect.ValueOf(any) //获取any的所有属性值
	fmt.Println(t,v)
}

func main() {
	CheckAny(Student{nikeName: "小张",age: 18})
}
main.Student {小张 18}

获取一个属性的值

func TypeInfo(any interface{})  {
	t := reflect.TypeOf(any)
	fieldCount := t.NumField() //获取any的属性数量

	v := reflect.ValueOf(any)
	for i := 0; i < fieldCount; i++ {
		fmt.Println(v.Field(i)) //通过 Value类型Field方法获取参数值,入参为属性索引
	}
}

获取深层次的属性值

type Student struct {
	nikeName string
	age int8
	Class struct{
		name string
	}
}


func GetField(any interface{}) {
	v := reflect.ValueOf(any)
	value := v.FieldByIndex([]int{0}) //通过一个二维切片,获取第几位属性中的第几位属性
	fmt.Println(value)
}

func main() {
	GetField(Student{nikeName: "小黑",age: 18,Class: struct{ name string }{name: "三年二班"}})
}
三年二班

通过Kind判断一个参数为什么类型

func ModifyMetaData(any interface{}){
	t := reflect.TypeOf(any)
	fmt.Println("any 是否是一个结构体:",t.Kind() == reflect.Struct)
	fmt.Println("any 是否是一个interface:",t.Kind() == reflect.Interface)
}

通过反射修改参数元数据

func ModifyMetaData(any interface{}){
	fmt.Println(any)
	v := reflect.ValueOf(any)
	elem := v.Elem()
	name := elem.FieldByName("NikeName") //获取nikeName字段
	if name.Kind() == reflect.String {
		name.SetString("反射修改的数据!")
	}
	fmt.Println(any)
}

func main() {
	ModifyMetaData(&Teacher{NikeName: "111"})
}
&{111 0}
&{反射修改的数据! 0}

通过反射执行结构体的方法

type Teacher struct {
	NikeName string
	age int8
}

func (Teacher) Run(str string)  {
	fmt.Println("hi say",str)
}

执行Run方法

func InvokeFunc(any interface{}) {
	v := reflect.ValueOf(any)
	count := v.NumMethod() 															//获取方法数量
	method := v.Method(0)                              //获得第一个方法
	str := []reflect.Value{reflect.ValueOf("我是参数")} //构造一个执行方法所需要的参数
	method.Call(str)																	//执行方法
}
