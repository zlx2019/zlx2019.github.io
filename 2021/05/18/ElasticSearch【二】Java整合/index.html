    User user1 = new User("王美丽","admin123",39,"南昌市","今天下雪了");

        System.out.println(item.getResponse().status());
    }
}

6. 查询文档列表
/**
 * 查询一个索引库的所有文档
 * @Author lx Zhang.
 * @Date 2021/5/18 3:27 下午
 **/
@Test
public void findDocuments() throws IOException {
    SearchRequest request = new SearchRequest("users"); //索引库名
    //构造查询条件,查询全部
    SearchSourceBuilder queryBuilder = new SearchSourceBuilder().query(QueryBuilders.matchAllQuery());
    request.source(queryBuilder);
    SearchResponse searchResponse = client.search(request, RequestOptions.DEFAULT);
    /** 遍历打印*/
    SearchHits searchHits = searchResponse.getHits();
    SearchHit[] hits = searchHits.getHits();
    for (SearchHit hit : hits) {
        System.out.println(hit.getSourceAsMap());
    }
}

    //构建bool多条件  title 中分词包含有 '服务' and price >19 and price < 100

9. 聚合查询
1.求最大值 max
/**
 * 聚合查询
 */
@Test
public void agg() throws IOException {
    SearchRequest searchRequest = new SearchRequest("books");
    SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
    // 构建聚合条件,聚合条件为 求price的最大值
    MaxAggregationBuilder aggregationBuilder = AggregationBuilders.max("priceMax").field("price");
    sourceBuilder.aggregation(aggregationBuilder);//设置聚合条件

    searchRequest.source(sourceBuilder);
    SearchResponse response = client.search(searchRequest, RequestOptions.DEFAULT);
    System.out.println(response.getHits());
    System.out.println(response.getTook());
    SearchHits hits = response.getHits();
    for (SearchHit hit : hits) {
        System.out.println(hit.getSourceAsMap());
    }
}

10. 条件构造案例
/**
 * Query 构建Demo
 * @Author lx Zhang.
 * @Date 2021/3/24 9:16 下午
 * @Param []
 * @return void
 **/
public void queryBuild(){
    SearchSourceBuilder builder = new SearchSourceBuilder();
    //构建查询所有条件
    builder.query(QueryBuilders.matchAllQuery());

    //构建多条件
    builder.query(QueryBuilders.boolQuery()
            .must(QueryBuilders.matchAllQuery())
            .must(QueryBuilders.termQuery("filed","value")));

    //分页 偏移量 -- 页容量
    builder.from(1).size(10);

    //排序
    builder.sort("filed", SortOrder.DESC);

    //filter 过滤器
    builder.postFilter(QueryBuilders.matchQuery("filed","value"));

    //高亮
    builder.highlighter(new HighlightBuilder().field("field").requireFieldMatch(false)
    .preTags("<span>").postTags("</span>"));
}
