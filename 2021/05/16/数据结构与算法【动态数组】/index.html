数据结构与算法【动态数组】

发表于2021-05-16|更新于2023-03-01|数据结构与算法
|字数总计:780|阅读时长:3分钟|阅读量:

前言

数组,是大多编程语言中的一个最基本也是最基础的一个数据结构,属于线性表,但是数组是不可以动态扩容的,也就是说在初始化数组时就已经决定了他的内存所占大小,那么就有了很大的局限性,不过众多语言都有它的强化版代表.比如Java中的List、Go中的切片等,为了更加深入的理解数组,本章将手写一个java中的切片(Slice)

话不多说直接上代码

package com.lxz.动态数组;

import java.util.Arrays;

/**
 * Java实现切片(动态数组)(灵感来源:ArrayList)
 * @author lx Zhang.
 * @date 2021/5/15 10:22 上午
 */
public class Slice<E> {

    /** ------------------------default value-----------------------------------------*/
    /** 默认初始化长度*/
    private static final int DEFAULT_SIZE_VALUE = 10;
    /** 自动扩容的最大长度*/
    private static final int DEFAULT_MAX_VALUE = Integer.MAX_VALUE - 8;



    /** ------------------------data Attributes-----------------------------------------*/
    /** 切片长度*/
    private int size;
    /** 数据存储数组*/
    private Object [] defaultValue;




    /** ------------------------construction-----------------------------------------*/

    /** 指定切片长度构造函数*/
    public Slice (int capacity){
        /** 指定的长度不能小于默认长度,否则将依然使用默认长度*/
        capacity = Math.max(capacity, DEFAULT_SIZE_VALUE);
        defaultValue = new Object[capacity];
    }
    /** 不指定长度则使用默认长度*/
    public Slice(){
        this(DEFAULT_SIZE_VALUE);
    }




    /** ------------------------------API-------------------------------------------*/

    /**
     * 获取当前切片扩容的长度
     * @return
     */
    public int count(){
        return this.defaultValue.length;
    }

    /**
     * 获取当前切片的数据数量
     * @return
     */
    public int size(){
        return this.size;
    }

    /**
     * 清空切片
     * 这里的清空只是重置了存储数组的长度,并没有清空数据,可以多次复用.
     */
    public void clear(){
        for (int i = 0; i < size; i++) {
            defaultValue[i] = null;
        }
        this.size = 0;
    }

    /**
     * 切片是否为空
     * @return true: empty
     *         false: not empty
     */
    public Boolean isEmpty(){
        return size == 0;
    }

    /**
     * 添加数据到尾部
     * @param value 泛型数据
     */
    public void add(E value){
        add(value,size);
    }
    /**
     * 添加数据到指定位置
     * @param value
     * @param index
     */
    public void add(E value,int index){
        if (index < 0 || index > size) throw new IndexOutOfBoundsException("index is max error!");

        reSet(size + 1);

        for (int i = size - 1; i >= index;i--){
            defaultValue[i + 1] = defaultValue[i]; //后移
        }
        defaultValue[index] = value;
        this.size++;
    }

    /**
     * 动态扩容1.5倍
     * @param capacity
     */
    private void reSet(int capacity) {
        int oldCapacity = defaultValue.length;
        if (oldCapacity >= capacity) return;
        int newCapacity = oldCapacity + (oldCapacity >> 1); //扩容1.5倍
        System.out.println(oldCapacity + "-" +newCapacity);
        Object[] newValues = new Object[newCapacity];
        for (int i = 0;i < size;i++){
            //将旧数组的数据,迁移到新数组
            newValues[i] = defaultValue[i];
        }
        defaultValue = newValues;
    }

    /**
     * 将新数据和指定的位置的数据替换,并且返回旧数据,
     * @param value
     * @param index
     * @return
     */
    public E replaceAndGet(E value,int index){
        verifyIndex(index);
        E oldValue = (E) defaultValue[index];
        defaultValue[index] = value;
        return oldValue;
    }


    /**
     * 根据索引获取数据
     * @param index
     */
    public E get(int index){
        verifyIndex(index);
        return (E) defaultValue[index];
    }


    /**
     * 根据索引删除并且返回数据
     * @param index
     * @return
     */
    public E  remove(int index){
        verifyIndex(index);
        E oldValue = (E) defaultValue[index];
        for (int i = index + 1;i <= size;i++){
            defaultValue[i - 1] = defaultValue[i];
        }
        defaultValue[--size] = null;
        return oldValue;
    }


    /**
     * 查看数据在切片中的索引
     * @param element
     * @return
     */
    public int indexOf(E element){
        for (int i = 0; i < size; i++) {
            if (defaultValue[i].equals(element)) return i;
        }
        return -1;
    }

    /**
     * 切片中是否包含该元素
     * @param element
     * @return
     */
    public boolean contains(E element){
        return indexOf(element) != -1;
    }

    /**
     * index索引 校验
     * @Author lx Zhang.
     * @Date 2021/5/15 2:30 下午
     * @Param [index]
     * @return void
     **/
    private void verifyIndex(int index){
        if (index < 0 || index >= size){
            throw new IndexOutOfBoundsException("index is out max size!");
        }
    }

    @Override
    public String toString() {
        return "Slice{" +
                "size=" + size +
                ", defaultValue=" + Arrays.toString(defaultValue) +
                '}';
    }
}
