Postgresql学习之旅【二】内置函数

发表于2021-05-05|更新于2021-05-06|Postgresql系列
|字数总计:923|阅读时长:3分钟|阅读量:

1. 数学函数

-- 获取绝对值
SELECT abs(-17.4) --R: 17.4

-- 立方根
SELECT cbrt(27) --R: 3

--大于参数的最小整数
SELECT ceil(55.8) --R: 56

--小于参数的最大整数
select floor(45.2) --R:45

--求两数相除的商
select div(9,4) --R: 2

--求两数的余数
select mod(9,4) --R: 1

--四舍五入 保留两位小数
SELECT round(65.4412, 2)  --R: 65.44

2.随机函数

-- 范围为0.0<= x < 1.0之中的随机数
SELECT random()

--随机获取小于100的整数
SELECT ceil(random() * 100)

3.字符串函数

--连接字符串
select 'Hello'||'Pgsql' --R:HelloPgsql

--连接字符串与非字符串
select '现在时间是:'|| now() --R:Hello2021-05-05 21:30:08.473724+08

--字符串包含的比特位数
select bit_length('H') --R: 8
select bit_length('好') --R: 24
select bit_length('H好') --R: 32

--字符串转换小写
select lower('HELLO') --R:hello

--转换大写
select upper('hello') --R:HELLO

--将所有的参数以文本形式连接起来
select concat('abcd',213,'真棒') --R:abcd213真棒

--字符串格式化,%1$s 表示继续使用第一个占位符
select format('Hello %s,is %1$s', 'World')

--将每个单词首字母转为大写,剩下字母转为小写,单词之间以空格区分
SELECT initcap('hELLO,WORLD') --R： Hello,World

--获取字符串的前 'n'个字符
select left('Hello',2) --R:He

--计算字符数的md5哈希值
select md5('HELLo wor') --R:ba50ee1efda092b2ae87aac268417c71

--字符串替换
select replace('lxz is max','lxz','zlx') --R:zlx is max

--将字符串重复n次 连接起来
SELECT repeat('lxz',4) --R: lxzlxzlxzlxz

--将字符串反转
select reverse('今天天气真不错') --R: 错不真气天天今

4.日期函数和时间函数

-- 两个日期相减,返回一个n年n月n日的结果
SELECT age( '2020-01-01',  '2000-05-01') --R: 19 years 8 mons

--当前日期和时间
SELECT clock_timestamp() --R: 2021-05-05 21:57:38.808704+08

--当前日期
select CURRENT_DATE

--当前时间
select CURRENT_TIME
select  LOCALTIME

--当前日期和时间
select CURRENT_TIMESTAMP
select LOCALTIMESTAMP

--使用年、月、日创建日期
select make_date(2020, 10, 20) --R:2020-10-20

--使用时、分、秒创建时间
select make_time(21, 30, 28) --R:21:30:28

--使用年、月、日、时、分、秒创建日期时间
select make_timestamp(2019, 6, 20, 22, 30, 00) --R:2019-06-20 22:30:00
--使用年、月、日、时、分、秒创建日期时间(带有时区,如果不填写时区,则为当前时区)
select make_timestamptz(2019, 6, 20, 22, 30, 00) --R:2019-06-20 22:30:00+08

--当前事务开始的时间
select now()

--当前sql语句开始的时间
SELECT statement_timestamp(),pg_sleep(1),statement_timestamp() --R:一致

5.聚合函数

Table:

root_db=# select * from func;
 num  
------
   90 
   21 
 3421 
   12 
   83 
   23 
(6 rows)
--将输入值,连接成一个数组
-- 将表中所有num连接成一个数组
SELECT array_agg(num) from func --R:{90,21,3421,12,83,23}

--获取平均值,并且四舍五入
select round(avg(num)) from func --R:608

--统计总结果行数量
select count(*) FROM func --R：6
--统计 参数为非空的行数
select count(num) FROM func --R:6

--获取字段最大值
select max(num) FROM func --R:3421
-- 获取最小值
select min(num) FROM func --R:12

--求和
SELECT sum(num) from func --R:3650

6.集合函数

-- 生成一系列值,从start-stop 步长为1
SELECT generate_series(1, 90) --R: 1-90

--生成一系列值,从1-90 步长为3
select generate_series(1, 90, 3)

7.系统信息函数

-- 获取当前数据库
select CURRENT_CATALOG
SELECT current_database()

--当前正在执行的sql
SELECT current_query()

-- 当前登陆者
SELECT CURRENT_USER
select CURRENT_ROLE

--当前模式
select "current_schema"()
select CURRENT_SCHEMA

-- 远程的客户端连接地址
select inet_client_addr()

--远程客户端连接端口
select inet_client_port()

--本地服务地址
select inet_server_addr()
select inet_server_port()

-- 与当前会话关联的服务器进程id
SELECT pg_backend_pid()

-- 数据库启动的时间
select pg_postmaster_start_time()

--数据库版本信息
select "version"()
