Netty配置类

/**
 * Netty 配置
 * @author lx Zhang.
 * @date 2021/7/5 2:38 下午
 */
@Configuration
public class NettyServerConfig {

    /**
     * 连接线程组
     * @return
     */
    @Bean
    public EventLoopGroup bossGroup(){
        return new NioEventLoopGroup();
    }

    /**
     * 工作线程组
     * @return
     */
    @Bean
    public EventLoopGroup workerGroup(){
        return new NioEventLoopGroup();
    }

    /**
     * websocket 事件处理类
     */
    @Autowired
    private ChatHandler chatHandler;
    @Autowired
    private NettyProperties nettyProperties;

    /**
     * Netty服务启动类
     * @param bossGroup 自动注入的连接线程组
     * @param workerGroup 自动注入的工作线程组
     * @return
     */
    @Bean
    public ServerBootstrap bootstrap(EventLoopGroup bossGroup,EventLoopGroup workerGroup){
        ServerBootstrap bootstrap = new ServerBootstrap();
        bootstrap.group(bossGroup,workerGroup) //设置线程组
                .channel(NioServerSocketChannel.class) //设置服务端通道类型
                .childHandler(new ChannelInitializer<SocketChannel>() {
                    @Override
                    protected void initChannel(SocketChannel ch) throws Exception {
                        ch.pipeline().addLast(new HttpServerCodec());// websocket 基于http协议，所以要有http编解码器
                        ch.pipeline().addLast(new ChunkedWriteHandler());// 对写大数据流的支持
                        ch.pipeline().addLast(new HttpObjectAggregator(65536));
                        ch.pipeline().addLast(new WebSocketServerProtocolHandler(nettyProperties.getWsPath())); //ws协议path
                        ch.pipeline().addLast(chatHandler);//注册聊天事件处理类
                    }
                }).option(ChannelOption.SO_BACKLOG,100)
                .childOption(ChannelOption.SO_KEEPALIVE,true); //开启心跳
        return bootstrap;
    }
}

事件处理类

/**
 * 聊天逻辑
 * @author lx Zhang.
 * @date 2021/7/5 3:55 下午
 */
@Component
@ChannelHandler.Sharable
@Slf4j
public class ChatHandler extends SimpleChannelInboundHandler<TextWebSocketFrame> {

    /** 用于存储连接的Channel*/
    private ChannelGroup channels = new DefaultChannelGroup(GlobalEventExecutor.INSTANCE);

    /**
     * 广播消息
     * @param ctx
     * @param msg
     * @throws Exception
     */
    @Override
    protected void channelRead0(ChannelHandlerContext ctx, TextWebSocketFrame msg) throws Exception {
        //获取客户端ip
        InetSocketAddress ipSocket = (InetSocketAddress)ctx.channel().remoteAddress();
        String clientAddress = ipSocket.getAddress().getHostAddress();
        String text = msg.text();
        channels.stream().forEach(channel -> {
            if (channel == ctx.channel()){
                channel.writeAndFlush(new TextWebSocketFrame("你说: " + text));
            }else {
                channel.writeAndFlush(new TextWebSocketFrame(clientAddress + ":" + text));
            }
        });
    }

    /**
     * 建立Ws连接事件
     * @param ctx
     * @throws Exception
     */
    @Override
    public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
        batchSendMessage(ctx.channel().remoteAddress() + "进入聊天室,大家快来看~");
        channels.add(ctx.channel());
    }

    /**
     * 退出事件
     * @param ctx
     * @throws Exception
     */
    @Override
    public void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
        batchSendMessage(ctx.channel().remoteAddress() + "退出聊天室~");
        channels.remove(ctx.channel());
    }

    /**
     * 广播消息
     */
    private void batchSendMessage(String message){
        channels.stream().forEach(channel -> {
            channel.writeAndFlush(new TextWebSocketFrame(message));
        });
    }
}

Netty启动类

/**
 * Netty启动类
 * @author lx Zhang.
 * @date 2021/7/5 3:27 下午
 */
@Slf4j
@Component
public class NettyApplication {

    @Autowired
    private NettyProperties nettyProperties;

    @Autowired
    private ServerBootstrap serverBootstrap;
    private Channel channel;

    /**
     * Netty 启动入口
     * @throws InterruptedException
     */
    public void start() throws InterruptedException {
        log.info("【Netty服务已启动】");
        channel = serverBootstrap.bind("0.0.0.0",nettyProperties.getPort()).sync().channel().closeFuture().sync().channel();
    }

    /**
     * Netty 停止服务
     */
    public void close(){
        channel.close();
        channel.parent().close();
    }
}

服务启动类

/**
 * 服务启动类
 */
@SpringBootApplication
@EnableConfigurationProperties(NettyProperties.class)
public class NettyChatApplication {

    public static void main(String[] args) {
        ConfigurableApplicationContext context = SpringApplication.run(NettyChatApplication.class, args);
        NettyApplication nettyApplication = context.getBean(NettyApplication.class);
        try {
            nettyApplication.start();//启动Netty
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>WebSocket Chat</title>
</head>
<body>
<script type="text/javascript">
    var socket;
    if (!window.WebSocket) {
        window.WebSocket = window.MozWebSocket;
    }
    if (window.WebSocket) {
        // socket = new WebSocket("ws://101.37.79.47:8888/websocket"); 服务器
        socket = new WebSocket("ws://localhost:8888/websocket");

        /** ws消息事件*/
        socket.onmessage = function(event) {
            var ta = document.getElementById('responseText');
            ta.value = ta.value + '\n' + event.data
        };

        /** 连接成功事件*/
        socket.onopen = function(event) {
            var ta = document.getElementById('responseText');
            ta.value = "连接开启!";
        };

        /** ws连接关闭事件*/
        socket.onclose = function(event) {
            var ta = document.getElementById('responseText');
            ta.value = ta.value + "连接被关闭";
        };
    } else {
        alert("你的浏览器不支持 WebSocket！");
    }

    function send(message) {
        if (!window.WebSocket) {
            return;
        }
        if (socket.readyState == WebSocket.OPEN) {
            socket.send(message);
            document.getElementById('message').value = '';
        } else {
            alert("连接没有开启.");
        }
    }
</script>
