Go【十四】优雅关闭http服务

发表于2023-02-11|更新于2023-03-01|Go
|字数总计:808|阅读时长:3分钟|阅读量:

优雅关闭Http服务

一个程序,如果强制进行关闭那么还在执行中的任务将会被强制打断,导致数据不一致、毁坏等情况,所以一个程序的退出,必须要安全且可靠。

Go体系中有许多框架或工具包都可以优雅的关闭Go Http服务,不过如果你使用的是Go1.8及以上版本,可以使用http.Server的Shutdonw()方法优雅关机。

下面会介绍两种优雅退出的方式,一种是基于channal,另一种是基于context,两种方式都可以实现。

这里实现gin+http的服务的优雅退出。

基于channal优雅退出

package main

import (
	"context"
	"github.com/gin-gonic/gin"
	"log"
	"net/http"
	"os"
	"os/signal"
	"syscall"
	"time"
)

func main() {
	// 创建gin引擎
	engine := gin.Default()
	// 通过原生httpServer + gin引擎开启服务
	server := &http.Server{
		Addr:    ":8080",
		Handler: engine,
	}
	// 这里通过一个协程来开启 http服务,保证服务能够主动性的优雅停止
	go func() {
		// 启动
		if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
			log.Fatalln(err)
		}
	}()
	// 由于服务由一个协程开启的,所以主协程这里我们需要手动阻塞一下,直到接受到停止信号
	// 创建一个接收操作系统信号的通道
	exit := make(chan os.Signal)
	// 这里表示如果接收到了SIGINT或者SIGTERM系统信号,则会把信号向exit通道发送.
	// syscall.SIGINT: 		用户发送INTR字符,例如在终端执行(Ctrl+C) 触发 kill -2 pid然后进程结束
	// syscall.SIGTERM: 	结束程序(可以被捕获、阻塞或忽略)
	signal.Notify(exit, syscall.SIGINT, syscall.SIGTERM)

	// 阻塞,直到接收到两种信号其中一种...
	<-exit

	// 信号接收到后,需要一定的时间释放相应的资源。 这里延迟3秒,模拟释放资源
	closeCtx, cancel := context.WithTimeout(context.Background(), time.Second*3)
	defer cancel()
	// 停止服务
	if err := server.Shutdown(closeCtx); err != nil {
		log.Println(err)
	}

	// 等待5秒时间过后,结束程序.
	select {
	case <-closeCtx.Done():
		log.Println("Wait Close Resource Timeout...")
	}
	log.Println("HTTP Server Shutdown Success")
}

基于context优雅退出

package main

import (
	"context"
	"github.com/gin-gonic/gin"
	"log"
	"net/http"
	"os/signal"
	"syscall"
	"time"
)

func main() {
	// 创建上下文,用于监听操作系统信号
	// 如果监听到SIGINT 或者 SIGTERM信号,那么该上下文则关闭 close(ctx)
	// syscall.SIGINT: 		用户发送INTR字符,例如在终端执行(Ctrl+C) 触发 kill -2 pid然后进程结束
	// syscall.SIGTERM: 	结束程序(可以被捕获、阻塞或忽略)
	exitCtx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	// 创建gin引擎
	engine := gin.Default()
	// 通过原生httpServer + gin引擎开启服务
	server := &http.Server{
		Addr:    ":8080",
		Handler: engine,
	}
	// 这里通过一个协程来开启 http服务,保证服务能够主动性的优雅停止
	go func() {
		// 启动
		if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
			log.Fatalln(err)
		}
	}()

	// 阻塞,直到接收到两种信号其中一种后释放
	<-exitCtx.Done()
	stop()

	// 信号接收到后,需要一定的时间释放相应的资源。 这里延迟3秒,模拟释放资源
	closeCtx, cancel := context.WithTimeout(context.Background(), time.Second*3)
	defer cancel()
	// 停止服务
	if err := server.Shutdown(closeCtx); err != nil {
		log.Println(err)
	}
	// 等待3秒时间过后,结束程序.
	select {
	case <-closeCtx.Done():
		log.Println("Wait Close Resource Timeout...")
	}
	log.Println("HTTP Server Shutdown Success")
}
