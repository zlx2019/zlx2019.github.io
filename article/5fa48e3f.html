Go【九】终端彩色输出

发表于2022-05-11|更新于2023-03-22|Go

前言

本章就是无聊搞点好玩的东西…嗯就这些。

终端彩色文本

通常我们在终端输出所观测到的都是黑白字,显得格外枯燥,在 Go中提供了一些文本颜色加成,本章就简单记录一下如何使用。

先来看一个输出格式:

fmt.Printf("\033[1;31;40m%s\033[0m\n","content.")
fmt.Printf("\033[1;37;41m%s\033[0m\n","message.")

其中除了指定的%s是文本内容外,其他都是格式。具体格式如下:

"\033[显示方式;字体颜色;背景颜色m%文本内容\033[0m"

下面简单测试一下:

fmt.Println("\033[1;31;1m这是要显示的文本内容\033[0m")
fmt.Println("\033[1;34;1m这是要显示的文本内容\033[0m")

参数如下:

  • 颜色相关

    注意: 字体颜色与背景颜色参数不一致。

    // 字体颜色 背景颜色 颜色
// ---------------------------------------
// 30  40  黑色
// 31  41  红色
// 32  42  绿色
// 33  43  黄色
// 34  44  蓝色
// 35  45  紫红色
// 36  46  青蓝色
// 37  47  白色

  • 显示方式

    // -------------------------
//  0  终端默认设置
//  1  高亮显示
//  4  使用下划线
//  5  闪烁
//  7  反白显示
//  8  不可见

下面是我定义好的库,可以直接使用

package utils

import (
	"fmt"
	"os"
	"strings"
	"time"
)

// 字体颜色常量,请参考上边列表
// 前景 背景 颜色
// ---------------------------------------
// 30  40  黑色
// 31  41  红色
// 32  42  绿色
// 33  43  黄色
// 34  44  蓝色
// 35  45  紫红色
// 36  46  青蓝色
// 37  47  白色

const (
	Black = iota + 30
	Rad
	Green
	Yellow
	Blue
	Magenta
	Cyan
	White
)

// 显示方式
// -------------------------
//  0  终端默认设置
//  1  高亮显示
//  4  使用下划线
//  5  闪烁
//  7  反白显示
//  8  不可见
const (
	Default    = 0
	High       = 1
	Underscore = 4
	Glint      = 5
	AntiWhite  = 7
	Invisible  = 8
)

// PrintlnCyan 将文本以Cyan颜色输出至控制台
func PrintlnCyan(message string) {
	fmt.Println(BaseHigh(message, Cyan))
}

// PrintlnRed 将文本以Red颜色输出至控制台
func PrintlnRed(message string) {
	fmt.Println(BaseHigh(message, Rad))
}

// PrintlnBlue 将文本以Blue颜色输出至控制台
func PrintlnBlue(message string) {
	fmt.Println(BaseHigh(message, Blue))
}

// BaseHigh 基础高亮语句
func BaseHigh(message string, textColor int) string {
	return SetColor(message, High, 1, textColor)
}

// SetColor 将字符特殊处理,返回后的文本可带有颜色高亮显示
func SetColor(message string, showType, backgroundColor, textColor int) string {
	return fmt.Sprintf("%c[%d;%d;%dm%s%c[0m", 0x1B, showType, textColor, backgroundColor, message, 0x1B)
}

// ProgressBar Go 模拟终端动态进度条
func ProgressBar() {
	j := 100
	for i := 0; i <= j; i++ {
		time.Sleep(50 * time.Millisecond)
		//拼接该次循环所需要展示的所有 = 和空白填充
		line := strings.Repeat("=", i) + strings.Repeat(" ", j-i)
		//计算进度
		completeTaskNum := float64(i) / float64(j) * 100
		fmt.Printf("\r%.0f% %[%s]", completeTaskNum, computeColor(line, i))
		os.Stdout.Sync()
	}
	fmt.Println()
	fmt.Println("load success.")
}

// 实时动态计算颜色值,产生一种彩色效果.
// message 渲染内容
// i 动态值
func computeColor(message string, i int) string {
	return fmt.Sprintf("%c[%d;%d;%dm%s%c[0m", 0x1B, 1, 0, 30+(i%7), message, 0x1B)
}

在 Mac 终端演示一下彩色输出及模拟进度条:

func main() {
	utils.PrintlnCyan("Hello World")
	utils.PrintlnRed("Hello World")
	utils.PrintlnBlue("Hello World")
	utils.ProgressBar()
}
