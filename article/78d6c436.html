Netty【一】实战Timer

发表于2021-07-04|更新于2023-03-22|Netty
|字数总计:1.1k|阅读时长:5分钟|阅读量:

前言

由于太久没有使用Netty,基本使用都快忘光了,今天本章就练练手,使用Netty实现一个网络通信案例获取服务器系统时间

服务端编写

package com.lxz.timer.server;

import io.netty.bootstrap.ServerBootstrap;
import io.netty.channel.ChannelFuture;
import io.netty.channel.ChannelInitializer;
import io.netty.channel.ChannelOption;
import io.netty.channel.EventLoopGroup;
import io.netty.channel.nio.NioEventLoopGroup;
import io.netty.channel.socket.SocketChannel;
import io.netty.channel.socket.nio.NioServerSocketChannel;
import io.netty.handler.codec.LineBasedFrameDecoder;
import io.netty.handler.codec.string.StringDecoder;

/**
 * Timer服务端
 * @author lx Zhang.
 * @date 2021/7/4 4:43 下午
 */
public class TimerServer {

    /**
     * Netty服务端启动方法
     * @Author lx Zhang.
     * @Date 2021/7/4 4:43 下午
     * @Param [port]
     **/
    public void bindServer(int port) throws Exception {
        //创建两个线程组 boss负责客户端连接事件,worker负责读写处理事件
        EventLoopGroup bossGroup = new NioEventLoopGroup();
        EventLoopGroup workerGroup = new NioEventLoopGroup();

        try {
            //创建服务并且配置相关参数
            ServerBootstrap application = new ServerBootstrap();
            application.group(bossGroup, workerGroup) //设置创建好的线程组
                    .channel(NioServerSocketChannel.class) //设置通道类型
                    .childHandler(new ChannelInitializer<SocketChannel>() {
                        @Override
                        protected void initChannel(SocketChannel socketChannel) throws Exception {
                            socketChannel.pipeline().addLast(new LineBasedFrameDecoder(1024)); //解决粘包问题,在响应数据后追加System.getProperty("line.separator")
                            socketChannel.pipeline().addLast(new StringDecoder()); //注册编码器
                            socketChannel.pipeline().addLast(new TimerServerHandler()); //注册服务端事件处理类
                        }
                    }).option(ChannelOption.SO_BACKLOG,1024) //最大队列数
                    .childOption(ChannelOption.SO_KEEPALIVE,true);//开启心跳

            //启动服务
            ChannelFuture future = application.bind(port).sync();
            future.channel().closeFuture().sync();
        } finally {
            bossGroup.shutdownGracefully();
            workerGroup.shutdownGracefully();
        }
    }

    public static void main(String[] args) throws Exception {
        new TimerServer().bindServer(8888);
    }
}

服务端事件处理

package com.lxz.timer.server;

import io.netty.buffer.ByteBuf;
import io.netty.buffer.Unpooled;
import io.netty.channel.ChannelHandlerContext;
import io.netty.channel.ChannelInboundHandlerAdapter;

import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;

/**
 * 服务端事件处理类
 * @author lx Zhang.
 * @date 2021/7/4 4:50 下午
 */
public class TimerServerHandler extends ChannelInboundHandlerAdapter {

    /**
     * 接收到客户端请求后的处理事件
     * 读取事件,获取服务端的时间并且响应给客户端
     * @Author lx Zhang.
     * @Date 2021/7/4 4:52 下午
     * @Param [ctx, msg]
     **/
    @Override
    public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
        String body = (String) msg; //获取客户端发送的IO消息
        //格式化当前系统时间
        String currentTime = DateTimeFormatter.ofPattern("yyyy年MM月dd日 hh:mm:ss").format(LocalDateTime.now());
        currentTime = "TIME ORDER".equalsIgnoreCase(body) ? currentTime : "ERROR TOKEN";

        ByteBuf response = Unpooled.copiedBuffer((currentTime + System.getProperty("line.separator")).getBytes());
        System.out.println("【接收到客户端请求,返回时间--" + currentTime);
        //将格式化后的时间写回客户端
        ctx.writeAndFlush(response);
    }

    @Override
    public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
        ctx.flush();
    }

    /**
     * 客户端建立连接事件
     * @Author lx Zhang.
     * @Date 2021/7/4 5:35 下午
     * @Param [ctx]
     **/
    @Override
    public void channelActive(ChannelHandlerContext ctx) throws Exception {
        System.out.println("【接收到客户端连接】"+ ctx.name());
    }

    @Override
    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
        ctx.close();
    }
}

客户端编写

package com.lxz.timer.client;

import io.netty.bootstrap.Bootstrap;
import io.netty.channel.ChannelFuture;
import io.netty.channel.ChannelInitializer;
import io.netty.channel.ChannelOption;
import io.netty.channel.EventLoopGroup;
import io.netty.channel.nio.NioEventLoopGroup;
import io.netty.channel.socket.SocketChannel;
import io.netty.channel.socket.nio.NioSocketChannel;
import io.netty.handler.codec.LineBasedFrameDecoder;
import io.netty.handler.codec.string.StringDecoder;

/**
 * Timer 客户端
 * @author lx Zhang.
 * @date 2021/7/4 5:05 下午
 */
public class TimerClient {

    /**
     * 客户端连接
     * @Author lx Zhang.
     * @Date 2021/7/4 5:10 下午
     * @Param [port, hostName]
     **/
    public void connect(int port,String hostName)throws Exception{
        EventLoopGroup clientGroup = new NioEventLoopGroup();
        try {
            Bootstrap client = new Bootstrap();
            client.group(clientGroup)
                    .channel(NioSocketChannel.class)
                    .option(ChannelOption.TCP_NODELAY,true)
                    .handler(new ChannelInitializer<SocketChannel>() {
                        @Override
                        protected void initChannel(SocketChannel socketChannel) throws Exception {
                            socketChannel.pipeline().addLast(new LineBasedFrameDecoder(1024));
                            socketChannel.pipeline().addLast(new StringDecoder()); //注册编码器
                            socketChannel.pipeline().addLast(new TimerClientHandler()); //注册客户端时间处理类
                        }
                    });

            ChannelFuture clientFuture = client.connect(hostName, port);
            clientFuture.channel().closeFuture().sync();
        } finally {
            clientGroup.shutdownGracefully();
        }
    }

    public static void main(String[] args) throws Exception {
        new TimerClient().connect(8888,"127.0.0.1");
    }
}

客户端事件处理

package com.lxz.timer.client;

import io.netty.buffer.ByteBuf;
import io.netty.buffer.Unpooled;
import io.netty.channel.ChannelHandlerContext;
import io.netty.channel.ChannelInboundHandlerAdapter;

/**
 * 客户端事件处理类
 * @author lx Zhang.
 * @date 2021/7/4 5:11 下午
 */
public class TimerClientHandler extends ChannelInboundHandlerAdapter {

    byte[] req;
    public TimerClientHandler() {
        req = ("TIME ORDER" + System.getProperty("line.separator")).getBytes();
    }

    /**
     * 客户端与服务端建立连接后回调的函数
     * @Author lx Zhang.
     * @Date 2021/7/4 5:15 下午
     * @Param [ctx]
     **/
    @Override
    public void channelActive(ChannelHandlerContext ctx) throws Exception {
        ByteBuf message = Unpooled.buffer(req.length); //开辟缓冲区
        message.writeBytes(req); //将消息填至缓冲区中
        ctx.writeAndFlush(message); //向服务端发送
    }


    /**
     * 接收服务端响应的事件
     * @Author lx Zhang.
     * @Date 2021/7/4 5:20 下午
     * @Param [ctx, msg]
     **/
    @Override
    public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
        String serverTimer = (String) msg;
        System.out.println("现在时间为:" + serverTimer);
    }


    @Override
    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
        ctx.close();
    }
}
