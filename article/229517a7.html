数据结构与算法【链表】

发表于2021-05-20|更新于2023-03-22|数据结构与算法

前言

上章,我们实现了自定义动态数组,但是动态数组有个明显的缺点,由于它的内存分配是连续的,如果没有将整个动态数组使用,那么将会存在内存空间的浪费,并且元素之间的增加、移除操作都会产生大量的移位操作,性能较差,那么我们怎么解决这些问题呢?链表,就可以解决这些问题,链表的内存是非连续分配,可以做到使用多少内存就分配多少内存

链表是一种链式存储线性表,所有元素的内存地址不一定是连续的

1.顶级接口

在Java中,接口是一种规范,或者说是一种约束,实现了该接口的类都将实现一系列指定的方法或者说操作,我们将链表、或者说动态数组公共的操作规范定义在接口中,对外提供

/**
 * 线性表顶级接口
 * @author lx Zhang.
 * @date 2021/5/20 9:47 上午
 */
public interface ListInterface<E> {

    /** 不存在*/
    Integer ELEMENT_NOT_FOUND = -1;

    /** 起始值 || 头节点索引*/
    Integer INDEX_ZERO = 0;

    /**
     * 清空链表
     */
    void clear();

    /**
     * 获取元素数量
     * @return 元素数量
     */
    int size();

    /**
     * 是否为空
     * @return true: 空
     */
    boolean isEmpty();

    /**
     * 是否存在该元素
     * @param element 判断的数据
     * @return true:存在
     */
    boolean contains(E element);

    /**
     * 添加元素到容器末尾
     * @param element 要添加的元素
     */
    void add(E element);

    /**
     * 添加元素到指定的位置
     * @param index 指定的位置
     * @param element 要添加的元素
     */
    void add(int index,E element);

    /**
     * 替换指定位置的元素,并且返回该位置替换前的元素
     * @param index 索引
     * @param element 新添加的元素
     * @return
     */
    E set(int index,E element);


    /**
     * 获取元素
     * @param index 索引
     * @return
     */
    E get(int index);

    /**
     * 移除元素
     * @param index 索引
     * @return
     */
    E remove(int index);

    /**
     * 获取元素在容器中的索引
     * @param element 元素
     * @return
     */
    int indexOf(E element);
}

2.抽象父类

接口是不允许有实现,但是我们发现链表和动态数组还是有共同实现点,所以将公共的方法实现,抽离到抽象父类中

/**
 * 线性表抽象父类,实现顶级接口,抽取出公共可用的方法
 * @author lx Zhang.
 * @date 2021/5/20 5:57 下午
 */
public abstract class AbstractLinked<E> implements ListInterface<E>{

    /** 容器元素的数量*/
    protected int size;


    /**
     * 获取元素数量
     *
     * @return 元素数量
     */
    @Override
    public int size() {
        return size;
    }

    /**
     * 是否为空
     *
     * @return true: 空
     */
    @Override
    public boolean isEmpty() {
        return size == 0;
    }


    /**
     * 是否存在该元素
     *
     * @param element 判断的数据
     * @return true:存在
     */
    @Override
    public boolean contains(E element) {
        return indexOf(element) != ELEMENT_NOT_FOUND;
    }

    /**
     * 添加元素到容器末尾
     *
     * @param element 要添加的元素
     */
    @Override
    public void add(E element) {
        add(size,element);
    }


    /** 索引校验,获取数据时,index不能给定大于或等于size的值,会越界*/
    protected void verifyIndexByGet(int index){
        if (index < 0 || index >= size){
            throwsException();
        }
    }

    /** 索引校验,添加数据时,index可以等于size,相当于在末尾添加数据,但是不能大于size*/
    protected void verifyIndexByAdd(int index){
        if (index < 0 || index > size){
            throwsException();
        }
    }

    /** 抛出索引越界异常*/
    protected void throwsException(){
        throw new IndexOutOfBoundsException("index is out max size...");
    }
}

3.链表节点元素类

/**
 * 链表元素节点
 * @author lx Zhang.
 * @date 2021/5/20 5:38 下午
 */
public class LinkedNode<E> {
    /** 数据*/
    E element;
    /** 下一个节点*/
    LinkedNode<E> next;

    public LinkedNode(E element, LinkedNode<E> next) {
        this.element = element;
        this.next = next;
    }
}

4.链表实现

/**
 * 自定义实现链表,继承自线性表抽象父类,实现属于自己的业务方法
 * @author lx Zhang.
 * @date 2021/5/20 9:29 上午
 */
public class LinkedTable<E> extends AbstractLinked<E>{

    /** 头节点*/
    LinkedNode<E> headNode;

    /**
     * 清除链表的所有元素,这里直接将头结点赋值为NULL,会断掉所有节点的引用,随后就会被GC回收掉
     */
    @Override
    public void clear() {
        size = 0;
        headNode = null;
    }

    /**
     * 添加元素到指定的位置
     *
     * @param index   指定的位置
     * @param element 要添加的元素
     */
    @Override
    public void add(int index, E element) {
        if (index == INDEX_ZERO){
            //添加到头结点情况要单独处理
            headNode = new LinkedNode<>(element,headNode);
        }else {
            LinkedNode<E> parentNode = findNodeByIndex(index - 1); //获取这个位置的父节点
            LinkedNode<E> newNode = new LinkedNode<>(element, parentNode.next);//创建新节点,并且将父节点的子节点赋值到自身
            parentNode.next = newNode; //将新节点连接到父节点上
        }
        size++; //容量+1
    }

    /**
     * 替换指定位置的元素,并且返回该位置替换前的元素
     *
     * @param index   索引
     * @param element 新添加的元素
     * @return
     */
    @Override
    public E set(int index, E element) {
        LinkedNode<E> node = findNodeByIndex(index);
        E oldEl = node.element; //获取旧元素
        node.element = element; //赋值新元素
        return oldEl;
    }

    /**
     * 获取指定位置的元素
     * @param index 索引
     * @return
     */
    @Override
    public E get(int index) {
        return findNodeByIndex(index).element;
    }

    /**
     * 移除元素
     *
     * @param index 索引
     * @return
     */
    @Override
    public E remove(int index) {
        E element = headNode.element;
        if (index == 0){//移除头节点
            headNode = headNode.next;
        }else{
            LinkedNode<E> parentNode = findNodeByIndex(index - 1); //获取要删除节点的父节点
            element = parentNode.next.element;
            if (null == parentNode.next.next){//移除末尾节点
                parentNode.next = null;
            }else{
                parentNode.next = parentNode.next.next;
            }
        }
        size--;
        return element;
    }

    /**
     * 获取元素在容器中的索引
     *
     * @param element 元素
     * @return -1: 表示不存在
     */
    @Override
    public int indexOf(E element) {
        LinkedNode<E> tempNode = headNode;
        int index = INDEX_ZERO - 1;
        do {
            index++;
            if (element.equals(tempNode.element)){
                break;
            }
            tempNode = tempNode.next; //后移
        }while (null != tempNode.next);
        /** 如果找到最后一个节点,并且最后一个节点也不是所需要的节点, 返回-1*/
        if (index == size - 1 && !element.equals(tempNode.element)){
            return  -1;
        }
        return index;
    }

    /**
     * 根据index索引,获取对应的节点
     * @Author lx Zhang.
     * @Date 2021/5/20 6:27 下午
     * @Param [index]
     * @return com.lxz.链表.LinkedNode<E>
     **/
    private LinkedNode<E> findNodeByIndex(int index){
        verifyIndexByGet(index); //校验索引
        LinkedNode<E> tempNode = this.headNode;//获取头节点
        for (int i = 0; i < index; i++) {
            if (tempNode == null){
                throw new RuntimeException("index exist error!");
            }
            tempNode = tempNode.next; //递归获取下一个节点,直到index次后
        }
        return tempNode;
    }

      /**
     * 是否是环形链表,利用快慢指针实现
     * @Author lx Zhang.
     * @Date 2021/5/21 3:05 下午
     * @return boolean
     **/
    public boolean hasCycle(){
        if (headNode == null || headNode.next == null) return false;
        LinkedNode<E> temp = headNode;
        LinkedNode slow = temp; //慢指针
        LinkedNode fast = temp.next; //快指针
        while (fast != null && fast.next != null){
            if (fast == slow) return true; //是环形链表
            slow = slow.next;
            fast = fast.next.next;
        }
        return false;
    }

    @Override
    public String toString() {
        StringBuilder builder = new StringBuilder();
        builder.append("size:" + size).append(",[");
        LinkedNode<E> tempNode = headNode;
        do {
            builder.append(tempNode.element).append(",");
            tempNode = tempNode.next;
        }while (null != tempNode);
        builder.append("]");
        return builder.toString();
    }
}
