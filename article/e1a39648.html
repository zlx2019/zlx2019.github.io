  "salary" int4 NOT NULL,

);

1.基本查询

  • 全表查询

    select * from employee;

  • 查询某些字段

    select empid,empname from employee;

  • 等值查询

    select * from employee where empid = 10001;

  • 时间范围查询

    select * from employee where createtime between to_date('2000-01-01', 'YYYY-MM-DD') AND to_date('2010-01-01', 'YYYY-MM-DD');

  • 去重,根据sex去重,男、女性别各留一个

    select distinct sex from employee;

  • 排序,DESC降序 ASC升序,默认ASC

    select * from employee ORDER BY createtime DESC;

  • 分组查询,根据性别分组,并且统计男女数量

    select sex,count(sex) from employee GROUP BY sex

  • 分组后再次筛选

    select sex,count(sex) from employee GROUP BY sex HAVING count(sex) > 10

  • 分页查询

    • limit : 查询页的总数量

    • offset :表示偏移量,跳过多少条数据再获取

      获取前5条数据

      select * from employee LIMIT 5 OFFSET 0

      每页3条数据,获取第2页

      select * from employee limit 3 offset 3 * (2-1)

      每页4条数据,获取第3页

      select * from employee limit 4 offset 4 * (3-1)

    • 由此得出,offset= （页容量 * (页码 - 1))

内连接

  • ON
    • ON字句是最常见的连接条件形式,它接收一个布尔值,一旦为ture,则他们就是匹配的行

  • USING

    • USING是一种快捷写法,在一种情况下十分方便,就是两个表连接的字段使用相同名称

  • 查询一年级的学生

    使用ON连接

    select u.*,e.name
from users u
INNER JOIN classe e
ON u.class_id = e.id
WHERE e.id = 10001

    使用USING, 注意,classe表的’id’,需要更改为’class_id’,与users的外键字段一致才行

    select u.*,e.name
FROM users u
INNER JOIN classe e
USING(class_id)

4.模糊查询

  • LIKE查询

    查询名字以 ‘小’起止的学生

    select * from users where name like '小%'

    查询名字中包含’小’的学生

    select * from users where name like '%小%'

    以’小’结尾的学生

    select * from users where name like '%小'

  • ILIKE

    忽略条件的大小写

    Like查询默认会区分大小写。

    select * from student where name like  'Ja%';
 id | name | score | class_id 
----+------+-------+----------
(0 rows)

    使用ILIKE表示不区分大小写

    select * from student where name ilike  'ja%';
 id | name | score | class_id 
----+------+-------+----------
  8 | jay  |    45 |     1001
(1 row)

    Pgsql提供了like简写:

    ~~ 等同于 like

    ~~* 等同于 ilike

    !~~ 等同于 not like

    !~~* 等同于 not ilike

    如下:

    select * from student where name ~~  'ja%';
select * from student where name ~~*  'ja%';

5.查询前n条记录

通过fetch first [count] rows only命令

select * from student order by score desc fetch first 2 rows only;
 id | name | score | class_id 
----+------+-------+----------
  6 | 小思 | 99.67 |     1001
  7 | 小璐 | 96.67 |     1001

count表示要查询的前几条记录数。

假如我们排序的字段,末尾记录有相同大小的记录,可以通过以下命令一并查询出来:

select * from student order by score desc fetch first 2 rows with ties; 
 id | name | score | class_id 
----+------+-------+----------
  6 | 小思 | 99.67 |     1001
  7 | 小璐 | 96.67 |     1001
 10 | 小l  | 96.67 |     1001

如果只是单纯查询前几条记录,可以通过limit [count]简化语句

my_db=# select * from student order by score desc  limit 2;
 id | name | score | class_id 
----+------+-------+----------
  6 | 小思 | 99.67 |     1001
  7 | 小璐 | 96.67 |     1001
(2 rows)

6.分页查询

offset 0 rows表示查询的起始偏移量,0表示第一条数据开始。

fetch first 3 rous noly 表示一页几条记录。

select * from student order by score desc offset 0 rows fetch first 3 rows only;
 id | name | score | class_id 
----+------+-------+----------
  6 | 小思 | 99.67 |     1001
  7 | 小璐 | 96.67 |     1001
 10 | 小l  | 96.67 |     1001
(3 rows)
select * from student order by score desc offset 3 rows fetch first 3 rows only;
 id | name | score | class_id 
----+------+-------+----------
  4 | 小赵 | 87.67 |     1001
  2 | 小李 | 64.67 |     1001
  8 | jay  |    45 |     1001
(3 rows)

分页同样支持简写

select * from student order by score desc limit 3 offset 0;
select * from student order by score desc limit 3 offset 3;

结果与上面两条一致.

7.分组查询

根据年级idclass_id进行分组,并且分组完再次经过条件筛选,统计的数据大于10。

select class_id, count(*) from student
group by class_id
having count(*) > 10
