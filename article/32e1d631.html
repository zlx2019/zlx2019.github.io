Vue3学习之旅【一】基础语法

前言

Vue3的学习日常笔记

1.基础插值语法

{{message}}<!--插值表达式-->
<!--表示不解析Vue的{{}}插值表达式,直接以string形式渲染-->
<p v-pre>{{message}}</p>
<!--只会渲染初始值,不会受双向绑定的影响,如果值进行改变,该标签不会再次渲染    -->
<p v-once>{{message}}</p>
<!--以普通文本形式渲染-->
<p v-text="message"></p>
<!--会解析成html形式渲染-->
<p v-html="message"></p>

2.属性绑定 v-bind:

    <!--{{}}的插值方式,无法在html标签的属性中使用,这样无法进行解析,表示'{{message}}'就是个字符串 如:-->
    <h2 title="{{message}}">hello</h2>

    <!--通过v-bind来进行标签属性绑定-->
    <h2 v-bind:title="title.message">{{message}}</h2>
    <!--v-bind: 简写语法糖如下    -->
    <h2 :title="title.message">{{message}}</h2>
    <img :src="img.url" :alt="img.message">
    
<!--#################################################################################-->
    
    <!--style行内样式的绑定 绑定数组的形式-->
    <div :style="['font-size:10px','background: red']">style bind</div>
    <br>
    <!-- 直接用对象绑定style-->
    <div :style="[style.font,style.background]">哈哈哈</div>
    style: {
        font: 'font-size:10px',
        background: 'background: red'
    }

<!--#################################################################################-->


    <!--绑定class属性-->
    <div :class="'customFont'">class</div><!--最基础的方式,基本不会用-->
    <!--绑定class 使用对象方式    -->
    <div :class="classes">class2</div>

    classes: 'customFont'

    <style scoped>
        .customFont{
            font-size: 20px;
        }
    </style>

    <!--动态绑定class-->
    <div :class="classes">绑定对象,对象关联style</div>
    data(){
        return{
            classes: ['customFont','customBackcolor'] //要绑定的class名称
    }}
    <style scoped>
        .customFont{
            font-size: 20px;
        }
        .customBackcolor{
            background-color: #42b983;
        }
    </style>
  • class绑定Example
    <!--动态绑定 class属性 Example-->
<template>
    <div>
        <!-- 绑定的对象都为boolean类型,并且名称和定义的style名称一致,如果为true,就会在该标签加上定义的style样式-->
        <div :class="{activeSty,isShowInfo}">
            真不错
        </div>
        <!--修改IsShowInfo的值-->
        <button v-on:click="isShowInfo = !isShowInfo">
            按钮
        </button>
    </div>
</template>

<script>
    export default {
        name: "App",
        data(){
            return{
                activeSty: true, //定义一个boolean值,名称与定义的class样式名称一样
                isShowInfo: true //隐藏属性
            }
        },
    }
</script>

<style scoped>
    /* 定义一个样式 */
    .activeSty{
        height: 100px;
        width: 100px;
        background: #42b983;
    }
    .isShowInfo{
        display: none;
    }
</style>

    3.计算属性 - computed

    computed 与 method的区别
  • computed 具有缓存效果,相同的业务多次运算,具有优势
  • computed 不是一个函数,无法带入参数,只有返回值
    <template>
  <div>
    <h2>{{getInfo}}</h2>
  </div>
</template>

<script>
export default {
  name: "Example4",
  data(){
    return{
      author: 'lx Zhang',
      message: 'Vue3初学者!'
    }
  },
  //计算属性,业务计算
  computed:{
    //将 author 与 message连接
    getInfo(){
      return this.author + ':' + this.message
    }
  }
}
</script>

4.事件监听 v-on

<!--Vue 事件监听-->
<template>
    <div>
        <!-- v-on绑定事件 -->
        <button :class="btnStyle" v-on:click="click">点击事件</button>
        <!--语法糖 @-->
        <button :class="btnStyle" @click="click($event)">点击事件</button>
    </div>
</template>

<script>
    export default {
        name: "App",
        data(){
            return{
                btnStyle: ['ui inverted blue button']
            }
        },
        methods:{
            click(){
                console.log("点击事件触发")
            }
        },
    }
</script>

5.事件修饰符

<!--stop 阻止冒泡 自己触发事件后,不会触发子元素的事件-->
<button :class="btnStyle" @click.stop="click">stop阻止冒泡</button>

<!--self 点击该元素本身才会触发事件,如果本身处于第二级元素,点击第一级元素不会触发该事件-->
<button :class="btnStyle" @click.self="click">self</button>

<!--capture 在冒泡中拥有更高的优先级 -->
<button @click.capture="click">capture</button>

<!--阻止元素本身的事件  只执行自己定义的事件  -->
<a href="https://101.37.79.47/" @click.prevent="click">lxz</a>

<!--once 只会触发一次事件    -->
<button @click.once="click"></button>

6.双向绑定 v-model

<!--通过表单元素的修改,同时也能修改data数据区域,就是双向绑定    -->
<input type="text" v-model="message">

<!-- v-model修饰符 懒加载  lazy:不会实时监听该数据,只有按enter键或者该输入框失焦后会进行再次绑定   -->
<input type="text" v-model.lazy="message">
<!--指定为数值类型    -->
<input type="text" v-model.number="message">
<!--自动去除两端字符串    -->
<input type="text" v-model.trim="message">
<br>

8.父子组件通信

  • 父组件

    <template>
  <div class="parent">
    父组件
    <br>
    子组件推送的数据:{{childData}}

    <!--通过 ':属性名' 向子组件传递参数,会传递到子组件的props属性中-->
    <!--定义@childPushEvent 事件,接收子组件推送的事件-->
    <Slave :names="names" @childPushEvent="listenerChild"></Slave>
  </div>
</template>

<script>
import Slave from "@/components/父子组件通信/Slave";

//父组件
export default {
  name: "Parent",
  components: {Slave},
  data(){
    return{
      names: 'lxz.vue',
      childData: '暂无'
    }
  },
  methods:{
    listenerChild(msg){
      console.log("接收到子组件推送的事件")
      this.childData = msg
    }
  }
}
</script>

<style scoped>
.parent{
  height: 200px;
  width: 2000px;
  background-color: antiquewhite;
}
</style>

  • 子组件

    <template>
  <div class="slave">
    子组件
    <br>

    父组件传递的数据:{{names}}
    <br>
    <button @click="push('这是子组件的数据')">推送数据到父组件</button>
    <br>
    <button @click="callParent">直接获取父组件数据</button>
  </div>
</template>

<script>
//子组件
export default {
  name: "slave",
  props:{
    //父组件传递的数据
    names:{
      type:String,
      default: "暂无"
    }
  },
  methods:{
    push(msg){
      //通过$emit('事件名',参数) 向父组件推送事件,childPushEvent事件必须在父组件声明
      this.$emit('childPushEvent',msg)
    },
    callParent(){
      //通过$parent 获取父组件
      console.log(this.$parent.names);
      this.$parent.showMsg()

      console.log('-------------')
      //通过$root 获取最上层父组件
      console.log(this.$root.names);
      this.$root.showMsg()
    }
  }
}
</script>

<style  scoped>
.slave{
  width: 1000px;
  height: 100px;
  background-color: #42b983;
}
</style>

9. Vue生命周期函数

<template>
  <div>
    {{msg}}
    <button @click="updateData('new data1')">修改数据</button>
    <button @click="updateData('new data2')">修改数据</button>
  </div>
</template>
<script>
export default {
  name: "App",
  data(){
    return{
      msg: "message"
    }
  },
  methods:{
    updateData(data){
      this.msg = data
    }
  },
  beforeCreate() {
    console.log('1-beforeCreate-组件实例刚刚初始化之后')
  },
  created() {
    console.log('2-created-组件实例已经创建完成')
  },
  beforeMount() {
    console.log('3-beforeMount-组件挂载DOM之前')
  },
  mounted() {
    console.log('4-mounted-组件挂载DOM完成')
  },
  beforeUpdate() {
    console.log('5-beforeUpdate-数据更新之前')
  },
  updated() {
    console.log('6-updated-数据更新')
  },
  activated() {
    console.log('被 keep-alive 缓存的组件激活时调用')
  },
  deactivated() {
    console.log('被 keep-alive 缓存的组件停用时调用。')
  },
  beforeUnmount() {
    console.log('7-beforeUnmount-卸载组件实例之前')
  },
  unmounted() {
    console.log('8-unmounted-卸载组件实例后')
  },
  errorCaptured(err,instance,info){
    console.log('捕获到子组件的错误')
    return true
  },
  renderTracked(event){
    console.log('跟踪虚拟DOM首次或重新渲染时调用,优先级高于mounted 或者 updated')
  },
  renderTriggered(event){
    console.log('此事件告诉你是什么操作触发了重新渲染，以及该操作的目标对象和键,此函数优先级比beforeUpdate要高')
  },
  beforeRouteLeave(to,from){
    console.log('从该组件路由到其他组件时触发的钩子函数')
  },
}
</script>
<style scoped></style>
9.1 beforeCreate
  • Type: Function
  • 详细
    • 在实例初始化之后，数据观测 (data observer) 和 event/watcher 事件配置之前被调用。
9.2 created
  • Type: Function
  • 详细
    • 在实例创建完成后被立即调用。在这一步，实例已完成以下的配置：数据观测 (data observer)，property 和方法的运算，watch/event 事件回调。然而，挂载阶段还没开始，$el property 目前尚不可用。
9.3 beforeMount
  • Type: Function
  • 详细
    • 在挂载开始之前被调用：相关的 render 函数首次被调用。该钩子在服务器端渲染期间不被调用。
9.4 mounted

  • Type: Function

  • 详细

    • 实例被挂载后调用，这时 Vue.createApp({}).mount() 被新创建的 vm.$el 替换了。如果根实例挂载到了一个文档内的元素上，当 mounted 被调用时 vm.$el 也在文档内。
    • 注意 mounted 不会保证所有的子组件也都一起被挂载。如果你希望等到整个视图都渲染完毕，可以在 mounted 内部使用 vm。$nextTick：

  • 使用:

    mounted() {
  this.$nextTick(function () {
    // 仅在渲染整个视图之后运行的代码
  })
}
9.5 beforeUpdate
  • Type: Function
  • 详细
    • 数据更新时调用，发生在虚拟 DOM 打补丁之前。这里适合在更新之前访问现有的 DOM，比如手动移除已添加的事件监听器。
    • 该钩子在服务器端渲染期间不被调用，因为只有初次渲染会在服务端进行。
9.6 updated

  • Type: Function

  • 详细

    • 由于数据更改导致的虚拟 DOM 重新渲染和打补丁，在这之后会调用该钩子。

    • 当这个钩子被调用时，组件 DOM 已经更新，所以你现在可以执行依赖于 DOM 的操作。然而在大多数情况下，你应该避免在此期间更改状态。如果要相应状态改变，通常最好使用计算属性或侦听器取而代之。

    • 使用

      updated() {
  this.$nextTick(function () {
    // 仅在渲染整个视图之后运行的代码
  })
}
9.7 activated
  • Type: Function
  • 详细
    • 被 keep-alive 缓存的组件激活时调用。
    • 该钩子在服务器端渲染期间不被调用。
9.8 deactivated
  • Type: Function
  • 详细
    • 被 keep-alive 缓存的组件停用时调用。
    • 该钩子在服务器端渲染期间不被调用。
9.9 beforeUnmount
  • Type: Function
  • 详细
    • 在卸载组件实例之前调用。在这个阶段，实例仍然是完全正常的。
    • 该钩子在服务器端渲染期间不被调用。
9.10 unmounted
  • Type: Function
  • 详细
    • 卸载组件实例后调用。调用此钩子时，组件实例的所有指令都被解除绑定，所有事件侦听器都被移除，所有子组件实例被卸载。
    • 该钩子在服务器端渲染期间不被调用。
9.11 errorCaptured
  • Type: (err: Error, instance: Component, info: string) => ?boolean
  • 详细
    • 当捕获一个来自子孙组件的错误时被调用。此钩子会收到三个参数：错误对象、发生错误的组件实例以及一个包含错误来源信息的字符串。此钩子可以返回 false 以阻止该错误继续向上传播。
  • 错误传播规则
    • 默认情况下，如果全局的 config.errorHandler 被定义，所有的错误仍会发送它，因此这些错误仍然会向单一的分析服务的地方进行汇报。
    • 如果一个组件的继承或父级从属链路中存在多个 errorCaptured 钩子，则它们将会被相同的错误逐个唤起。
    • 如果此 errorCaptured 钩子自身抛出了一个错误，则这个新错误和原本被捕获的错误都会发送给全局的 config.errorHandler。
    • 一个 errorCaptured 钩子能够返回 false 以阻止错误继续向上传播。本质上是说“这个错误已经被搞定了且应该被忽略”。它会阻止其它任何会被这个错误唤起的 errorCaptured 钩子和全局的 config.errorHandler。
9.12 renderTracked

  • Type: (e: DebuggerEvent) => void

  • 详细

    • 跟踪虚拟 DOM 重新渲染时调用。钩子接收 debugger event 作为参数。此事件告诉你哪个操作跟踪了组件以及该操作的目标对象和键。

  • 用法

    <div id="app">
  <button v-on:click="addToCart">Add to cart</button>
  <p>Cart({{ cart }})</p>
</div>
    const app = Vue.createApp({
  data() {
    return {
      cart: 0
    }
  },
  renderTracked({ key, target, type }) {
    console.log({ key, target, type })
    /* 当组件第一次渲染时，这将被记录下来:
    {
      key: "cart",
      target: {
        cart: 0
      },
      type: "get"
    }
    */
  },
  methods: {
    addToCart() {
      this.cart += 1
    }
  }
})

app.mount('#app')
9.13 renderTriggered

  • Type: (e: DebuggerEvent) => void

  • 详细

    • 当虚拟 DOM 重新渲染为 triggered.Similarly 为renderTracked，接收 debugger event 作为参数。此事件告诉你是什么操作触发了重新渲染，以及该操作的目标对象和键。

  • 用法

    <div id="app">
  <button v-on:click="addToCart">Add to cart</button>
  <p>Cart({{ cart }})</p>
</div>
    const app = Vue.createApp({
  data() {
    return {
      cart: 0
    }
  },
  renderTriggered({ key, target, type }) {
    console.log({ key, target, type })
  },
  methods: {
    addToCart() {
      this.cart += 1
      /* 这将导致renderTriggered调用
        {
          key: "cart",
          target: {
            cart: 1
          },
          type: "set"
        }
      */
    }
  }
})

app.mount('#app')
