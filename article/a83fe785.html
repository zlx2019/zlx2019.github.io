Mysql内置函数

1. 数学函数

获取绝对值
select ABS(0.8),ABS(-0.8) #返回绝对值 8,8

向上取整
SELECT CEILING(4.1),CEIL(-2.1) #返回大于该值的最小整数 5,-2

向下取整
SELECT FLOOR(4.1),FLOOR(-5.1) #返回小于或等于该值的最大整数 4,-6

取余数的三种方式
SELECT MOD(10,3),10 % 3, 10 MOD 3 #取余数,返回10除以3后的余数

四舍五入 不设置精度,则为整数
SELECT ROUND(1.7823,2) #将数值四舍五入,并且保留2位小数

截取数值,保留指定的精度
select TRUNCATE(1.999,1) #保留一位小数,其他直接舍弃
select TRUNCATE(1.999,0) #舍弃掉小数

获取随机小数,小于1
select RAND(),RAND()

example：随机获取一个100以内的整数  #三种实现
select ROUND(RAND() * 100) as num1,TRUNCATE(RAND() * 100,0)as num2,CEIL(RAND() * 100) as num3

example: 随机获取一个指定的范围内的整数 
获取 R, 条件: 15 <= R < 50
SELECT FLOOR(15 + RAND() * (50 - 15))

2.聚合函数

#count 统计表中数据数量,通常会根据表中的 主键统计
select count(user_id) from tb_salary

#sum 统计某一列数据总和
select sum(salary) FROM tb_salary

#avg 统计某一列数据的平均值
select avg(salary) from tb_salary

#max min 统计最高薪资和最低薪资
select max(salary) as max_salary,min(salary) as min_salary from tb_salary

#DISTINCT 去重统计,将数值一致的数据抹除掉,只保留一条
select count(DISTINCT salary) from tb_salary


###example
#GROUP_CONCAT的使用 将uint相同的数据列, 将某个字段连接为一个字段,并且以uint分组
#Tables:
+------+------------+
| uint | date       |
+------+------------+
| a    | 2020-09-11 |
| a    | 2021-04-11 |
| b    | 2021-09-09 |
| b    | 2021-04-06 |
| c    | 2021-04-11 |
| c    | 2013-06-11 |
| d    | 2021-04-11 |
+------+------------+
#Sql:
select uint,GROUP_CONCAT(`date`) from test_count GROUP BY uint

#Result:
+------+-----------------------+
| uint | GROUP_CONCAT(`date`)  |
+------+-----------------------+
| a    | 2020-09-11,2021-04-11 |
| b    | 2021-09-09,2021-04-06 |
| c    | 2021-04-11,2013-06-11 |
| d    | 2021-04-11            |
+------+-----------------------+
4 rows in set (0.00 sec)


# HAVING的使用 在分组之后,再次进行条件筛选
select COUNT(user_id),salary from tb_salary GROUP BY salary HAVING salary IS NOT NULL

注: 聚合函数无法嵌套使用,如:

select count(max(salary))
example: 按照年份、日统计,并且以unit分组

select
    uint as '单位',
    count(TO_DAYS(`date`) = TO_DAYS(NOW()) or NULL) as '今日统计',
    count(YEAR(`date`) = YEAR(NOW()) or null) as '今年统计'
from test_count
group by uint

3.字符串函数

  • 字符串大小写转换
    # 转换小写
select LOWER('MYSQL') #Result: mysql
# 转换大写
select UPPER('mysql') #Result: MYSQL
  • 多个字符串连接
    select CONCAT('M','Y','SQL') #Result: MYSQL

如果有任何一个参数为null,则结果为null
select CONCAT('M',null,'SQL') #Result: null

如果是数字,则自动转换为字符串
select CONCAT('M','Y','SQL',5.8) #Result: MYSQL5.8
  • 根据指定的连接符,进行字符串拼接
    select CONCAT_WS('-','Python','Java','Golang') #Result: Python-Java-Golang
  • 字符串截取
    # 字符串截取,从下标为5的字符开始截取 mysql 字符下标从1开始
select substring('hello,word',5) #Result: o,word

# 字符串截取,从下标为5开始截取3位
select substring('hello,word',5,3) #Result: o,w
  • 字符串统计
    # 获取字符串的所占字节 utf8:一个中文占3字节 gbk:一个中文占2个字节
select LENGTH('hello,word'),LENGTH('你好啊') #Result: 10, 9

# 获取字符串的字符数量
select char_length('Java'),CHAR_LENGTH('你好啊') #Result: 4,3
  • 在一个字符串中查找 某个字符串第一次出现的位置
    select INSTR('foobaerbak','ba') #Result: 4
  • 字符填充
    # 字符串填充,在源字符串的基础上,如果未到达指定的字节上限,将用 ‘*’ 从字符头部补充字节 
select LPAD('source','10','*') #Result: ****source

#从字符串尾部填充
select rpad('source','10','*') #Result: source****
  • 字符串剔除
    #去除两端空格 默认是去除空格
select TRIM('  hello  ') #Result:hello

# 去除头部 指定的字符
select TRIM(leading ';' FROM ';;;hello;;;') #Result: hello;;;

# 去除尾部 指定的字符
select TRIM(TRAILING ';' FROM ';;;hello;;;') #Result: ;;;hello

#去除头部、尾部 指定的字符
select TRIM(both ';' FROM ';;;hello;;;') #Result: hello
  • 字符串替换
    # 字符串替换
select REPLACE('www.mysql.com.com','com','cn') #Result: www.mysql.cn.cn
  • 字符串反转
    select REVERSE('ABCDEFG') #Result: GFEDCBA

4. 日期时间函数

#获取当前日期(不包含时间)
SELECT CURDATE() #Result: 2021-04-14
#获取当前时间(不包含日期)
select CURTIME() #Result: 17:54:27
#获取时间戳 (秒级时间戳)
select UNIX_TIMESTAMP() #Result: 1618394363
#获取当前日期和时间
select NOW() #Result: 2021-04-14 17:57:17
#类似now()的还有
select CURRENT_TIMESTAMP(),LOCALTIMESTAMP(),LOCALTIME()
#注意: 
#systemdate() 返回的是执行到该函数的时间,多次调用该函数之前的间隔时间较大的话,获取的时间将会不一致
#now() 返回的是该Sql语句执行的时间,无论一条sql中调用几次now(),返回的时间都是一致的
#例：
select NOW(),SLEEP(2),NOW() 				#Result: 2021-04-14 18:05:21	0	2021-04-14 18:05:21
select SYSDATE(),SLEEP(2),SYSDATE() #Result: 2021-04-14 18:06:03	0	2021-04-14 18:06:05
# 在以上情况下使用sysdate()会处在一定的问题,所以请使用now()

# 获取两个时间的差值 expr1 - expr2
select TIMEDIFF('12:30:00','8:00:00') #Result: 04:30:00
# 获取两个日期相差的天数
select DATEDIFF('2020-01-01','2000-01-01') #Result 7305
# 日期添加 在指定的日期上添加一天
select DATE_ADD(NOW(),INTERVAL 1 DAY)
#减一天
select DATE_SUB(NOW(),INTERVAL 1 day)
#在指定的日期时间上 减去 1天2两小时30分钟
select date_sub('2020-01-05 12:30:00',INTERVAL '1 2:30:00' DAY_SECOND)

#获取当前星期几、几月份
set lc_time_names='zh_CN' #设置中文
select DAYNAME(NOW()),MONTHNAME(NOW()) #Result: 星期三 ,四月

5.加密函数

#MD5加密(不可逆)
select MD5('lxz') #Result: 5d52a67facac487d8aa43f29de250510
select sha('lxz') #Result: 3f8306b511c633d761f5e0fc9386b791bb028fbb
select SHA1('lxz')#Result: 3f8306b511c633d761f5e0fc9386b791bb028fbb

#AES对称加密 AES_ENCRYPT(要加密的数据,加密的秘钥值), to_base64()进行base64编码
select to_base64(AES_ENCRYPT('lxz','keys')) #Result: eT8ZPEfQwax8t/W5gY4yfA==
#AES对称解密 AES_DECRYPT(要解密的数据,加密时的秘钥值) from_base64()进行base64解码
select AES_DECRYPT(from_base64('eT8ZPEfQwax8t/W5gY4yfA=='),'keys') #Result: lxz

6.流程控制

# 多条流程控制sql case: 选择的流程分支,如果是1,则会执行 when1 分支, 如果是2,则会执行 when2 分支,其他情况都会走else分支
select 	case 1
when 1 
	then 'one'
when 2 
	then 'two'
else 'more' end;

#查询用户状态
select case `status` 
	when 1
		then '已删除'
	when 2
		then '已冻结'
	else '正常' end
from user

#IF 语句,expr1=true,会执行expr2,反之执行expr3
#IF(expr1,expr2,expr3)
select IF(1 < 2,'Result_true','Result_false') #Result: Result_true
select IF(1 > 2,'Result_true','Result_false') #Result: Result_false

#IFNULL 语句,expr1如果不为NULL,就返回expr1,反之返回expr2
#IFNULL(expr1,expr2)
select IFNULL(1,123) #Result: 1
select IFNULL(NULL,123) #Result: 123
select IFNULL(0,123) #Result: 0


#NULLIF 语句,如果expr1==expr2,则返回NULL,反之返回expr1
#NULLIF(expr1,expr2)
select NULLIF(1,1) #Result: NULL
select NULLIF('first','last') #Result: first

7.格式化函数

# 日期时间格式化
# 日期时间格式化
select date_format(NOW(),'%Y %r') #Result: 2021 07:11:10 PM
select date_format(NOW(),'%Y-%m-%d') #Result: 2021-04-19
select date_format(NOW(),'%Y-%m-%d %H:%i:%s') #Result: 2021-04-19 19:14:13
#反格式化 日期
select date_format(19990306,'%Y-%m-%d') #Result: 1999-03-06
select date_format(19990306122530,'%Y-%m-%d %H:%i:%s') #Result: 1999-03-06 12:25:30

#获取当前是AM(上午)还是下午(PM)
select DATE_FORMAT(NOW(),'%p') 
#结合case使用
select case DATE_FORMAT(NOW(),'%p')
	WHEN 'AM'
		THEN '现在是上午'
	WHEN 'PM'
		THEN '现在是下午'
	ELSE '未知' END
